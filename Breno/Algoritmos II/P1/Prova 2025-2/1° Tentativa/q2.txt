funcao caracteresApartirDeC (c: caracter ; n: inteiro): string
var
	s: string
	aux: caracter
	i: inteiro
inicio
	s <- ""
	i <- 1
	aux <- c
	se n > 0 entao
		enquanto i <= n e aux >= 32 e aux <= 126 faça
			s <- s + chr(aux)
			aux <- aux + 1
			i <- i + 1
		fim enquanto
	senao
		enquanto i <= n e aux >= 32 e aux <= 126 faça
			s <- s + chr(aux)
			aux <- aux - 1
			i <- i + 1
		fim enqunato
	fim se
	retorne s
fim funcao
