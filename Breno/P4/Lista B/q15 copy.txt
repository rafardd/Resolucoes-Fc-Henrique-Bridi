algoritmo BuscarOValorMaisProximoDaMedia
var
    i: inteiro
    encontrou_empate: logico
    media, menordist,valor1,valor2,dist: real
    A: vetor[1..537] de real
inicio
    media <- 0
    para i de 1 até 537 faça
        escreva("Insira o ", i,"° número: ")
        leia(Ai])
        media <- media + A[i]
    fim para
    media <- media/537
    menordist <- abs(A[1] - media)
    valor1 <- A[1]
    encontrou_empate <- FALSO
    para i de 2 até 537 faça
        dist <- abs(A[i]-media)
        se menordist > dist e valor1 <> A[i] entao
            menordist <- dist
            valor1 <- A[i]
            encontrou_empate <- FALSO
        senao
            se dist = menordist entao
                valor2 <- A[i]
                encontrou_empate <- VERDADEIRO
            fim se
        fim se
    fim para
    se encontrou_empate entao
        escreva("O valor menor mais proximo é ",valor1," e o valor maior mais proximo é ", valor2)
    senao
        escreva("O valor mais proximo é ",valor1)
    fim se
fim algoritmo
