algoritmo LerNNumerosReais
var
    v: vetor[1..n] de real
    n, i: inteiro
inicio
    escreva("Insira um numero entre 1 e 111: ")
    leia(n)
    enquanto n < 0 e n > 111 faca   
        escrev("Numero invalido. Tente novamente: ")
        leia(n)
    fim enquanto
    para i de 1 até n faça
        escreva("Insira o ",i,"° número real: ")
        leia(v[i])
    fim para
    i <- n
    para i de 1 até n faça
        escreva(v[1-n+i])
    fim para
fim algoritmo
