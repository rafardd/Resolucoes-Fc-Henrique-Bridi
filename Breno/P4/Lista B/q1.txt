algoritmo VerificarDuasSequenciasDeNumerosSeSaoIguais
const
    MAX = 543
tipo
    QuantidadeUsuario: inteiro
    PrimeiraSequencia: vetor[1..n] de inteiro
    SegundaSequencia: vetor[1..n] de inteiro
    Contador: inteiro
    Flag: logico
var
    n : QuantidadeUsuario
    v1: PrimeiraSequencia
    v2: SegundaSequencia
    i: inteiro
    é_igual: Flag
inicio 
    escreva("Insira um numero menor que 543: ")
    leia(n)
    enquanto n >= MAX faça // RESTRIÇÃO DE MAXIMO
        escreva("Numero não permitido. Insira novamente: ")
        leia(n)
    fim enquanto
    para i de 1 até n faça // PRIMEIRA SEQUENCIA
        escreva("Primeira sequencia: Insira o ", i,"° número: ")
        leia(v1[i])
    fim para
    para i de 1 até n faça // SEGUNDA SEQUENCIA
        escreva("Segunda sequencia: Insira o", i,"° número: ")
        leia(v2[i])
    fim para
    é_igual <- VERDADEIRO
    i <- 1
    enquanto i <= n e é_igual faça // VERIFICAR SE SÃO IGUAIS
        se v1[i] <> v2[i] entao
            é_igual <- é_igual e FALSO
        fim se
        i <- i + 1
    fim enquanto
    se é_igual entao
        escreva("As duas sequencias são iguais.")
    senao
        escreva("Não são iguais.")
    fim se
fim algoritmo
