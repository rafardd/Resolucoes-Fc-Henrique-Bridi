algoritmo RetiraEspaçosEmBranco
var
    frase, frase_modificada: string
    i: inteiro
inicio
    escreva("Insira uma frase: ")
    leia (frase)
    frase_modificada <- "" // String Vazia para fazer a concatenacao
    para i de 1 até tamanho(frase) faça
        se frase[i] <> " " entao // Ver se o caracter é diferente de space
            frase_modificada <- frase_modificada + frase[i]
        fim se
    fim para
    escreva(frase_modificada)
fim algoritmo
