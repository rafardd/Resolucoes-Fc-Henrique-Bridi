algoritmo VerificarCaracteresAlternados
var
    s: string
    c, c2: caracter
    i: inteiro
    prox_maior, atual_maior, alterna: logico
inicio
    escreva()
    leia(s)
    alterna <- VERDADEIRO
    alterna <- alterna e (tamanho(s) > 0) // PORQUE AI SE O TAMANHO FOR MAIOR QUE 0 É POSSIVEL FAZER A VERIFICACAO DE ALTERNANCIA
    se tamanho(s) > 1 entao
        i <- 1
        alterna <- alterna e s[i] <> s[i+1] // VERIFICAR SE OS DOIS CARACTERES PROXIMOS NÃO SÃO IGUAIS
        maior_menor <- s[i] > s[i+1] // PARA SABER O PROXIMO CARACTER SE VAI SER MAIOR OU MENOR, SENDO MESMO ESQUEMA
                                     // VERDADEIRO PARA O CARACTER ADIANTE SER MAIOR
                                     // FALSO PARA O CARACTER ADIANTE SER MENOR
        i <- 2 // COMEÇAR NO SEGUNDO CARACTER POIS ELE É O MAIS RECENTE
        enquanto i < tamanho[s] faça
            se maior_menor entao // SE DEU VERDADEIRO É PORQUE O PROXIMO TEM QUE SER MAIOR
                alterna <- alterna e s[i] < s[i+1] 
            senao // SE NAO DEU VERDADEIRO É PORQUE O PROXIMO TEM QUE SER MENOR
                alterna <- alterna e s[i] > s[i+1]
            fimse
            maior_menor <- nao maior_menor // ISSO AQUI SERVE PARA FICAR ALTERNANDO ENTRE MAIOR MENOR MAIOR....
        fim enquanto
    fim se
    se alterna entao
            escreva("Sucesso")
    senao
        escreva("Falha")
    fim se
fim algoritmo
