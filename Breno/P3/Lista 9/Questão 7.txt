algoritmo EspaçosDuplicadosEmSimples
var
    s, s_nova: string
    i: inteiro
inicio
    escreva("Insira uma frase: ")
    leia(s) = ___Eu__Amo________A_Isadora___
    s_nova <- ""
    i <- 1
    enquanto s[i] = " " faça // Tirar todos os espaços do começo
        i <- i + 1
    fim enquanto
    enquanto i + 1 <= tamanho(s) faça // ELE PARA NO PENULTIMO PARA NAO VERIFICAR O I FINAL+ 1
        e_spaco_multiplo <- s[i] = " " e s[i+1] = " " // se for e multiplo entao nao adiciona
        se nao e_spaco_multiplo entao
            s_nova <- s_nova + s[i]
        fim se
        i <- i + 1
    fim enquanto
    se s[tamanho(s)] <> " " entao // SE O ULTIMO NAO FOR ESPAÇO ENTAO ADICIONA
        s_nova <- s_nova + s[tamanho(s)]
    fimse
    escreva (s_nova)
fim algoritmo
