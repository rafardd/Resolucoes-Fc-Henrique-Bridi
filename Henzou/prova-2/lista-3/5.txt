função retornaApariçõesDeBemA(a, b : inteiro): inteiro
início
  se a < 0 ou b < 0 ou b > 9 então
    retorne -1
  fimse

  se a < 10 então
    se a = b entao
      retorne 1
    senão
      retorne 0
    fimse
  fimse
  
  se a mod 10 = b entao
    retorne 1 + retornaApariçõesDeBemA(a div 10, b)
  senão
    retorne retornaApariçõesDeBemA(a div 10, b)
  fimse
fimfunção
