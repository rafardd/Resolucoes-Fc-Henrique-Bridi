função escreveValoresDeSomatório(n : inteiro): inteiro
var
  somaAtual, somaAnterior : inteiro
início
  se n = 0 então
    escreva(0)
    retorne 0
  senão
    somaAnterior <- escrevaValoresDeSomatório(n - 1)
    somaAtual <- n + somaAnterior
    escreva(", ", somaAtual)
    retorne somaAtual
  fimse
fimfunção
