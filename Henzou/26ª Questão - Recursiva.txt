tipos
    PNo : ponteiro para TNo
    TNo : registro
            info : real
            prox : PNo
          fimregistro

função somaNós(ref l: PNo; p: inteiro): lógico
var
    temp : PNo
início
    se l = NULO ou l^.prox = NULO então
        retorne FALSO
    fimse

    se p = 1 então
        temp <- l^.prox
        l^.info <- l^.info + temp^.info
        l^.prox <- temp^.prox
        desaloca(temp)
        retorne VERDADEIRO
    fimse
    retorne somaNós(l^.prox, p-1)
fimfunção
