tipos
    PNo = ponteiro para TNo
    TNo = registro
            chave : inteiro
            info : real
            esq : PNo
            dir : PNo
          fimregistro

procedimento inverte(ref l: PNo)
var
    aux : PNo
início
    se l != NULO então
        aux <- l^.esq
        l^.esq <- l^.dir
        l^.dir <- aux

        Inverte(l^.esq)
        Inverte(l^.dir)
    fimse
fimprocedimento
