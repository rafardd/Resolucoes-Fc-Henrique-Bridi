tipos
    PNo = ponteiro para TNo
    TNo = registro
            info : chave
            prox : PNo
          fimregistro

função comparaElementos(l1, l2 : PNo): lógico
var
    aparece, iguais : lógico
    cópial2 : PNo
início
    cópial2 <- l2
    iguais <- VERDADEIRO
    enquanto l1 != NULO faça
        l2 <- cópial2
        aparece <- FALSO
        enquanto l2 != NULO faça
            aparece <- aparece ou l1^.info = l2^.info
            l2 = l2^.prox
        fimenquanto
        iguais <- iguais e aparece
        l1 = l1^.prox
    fimenquanto
    retorne iguais
fimfunção
