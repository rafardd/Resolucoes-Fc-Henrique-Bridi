tipos
    PNo = ponteiro para TNo
    TNo = registro
            info : real
            prox : PNo
          fimregistro

função procuraElemento(l: PNo; X: real): lógico
início
    enquanto l != NULO e l^info != X faça
        l <- l^.prox
    fimenquanto
    se l = NULO então
        retorne FALSO
    senão
        retorne VERDADEIRO
    fimse
fimfunção
