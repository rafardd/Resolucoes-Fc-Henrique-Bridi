tipos
    PNo = ponteiro para TNo
    TNo = registro
            info : real
            prox : PNo
          fimregistro

função verificaElementosEmOrdem(l: PNo; X, Y : real): lógico
var
    AchouX, AchouY : lógico
início
    AchouX <- FALSO
    AchouY <- FALSO
    enquanto l != NULO faça
        AchouX <- AchouX ou l^.info = X
        se l^.prox != NULO então
            se l^.prox^.info = Y e AchouX então
                AchouY <- VERDADEIRO
            senão
                AchouX <- FALSO
            fimse
        fimse
        l = l^.prox
    fimenquanto

    se AchouX e AchouY então
        retorne VERDADEIRO
    senão
        retorne FALSO
    fimse
fimfunção
