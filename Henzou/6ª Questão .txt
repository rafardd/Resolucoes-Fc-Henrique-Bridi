tipos
    PNo : ponteiro para TNo
    TNo : registro
            info : string
            prox : PNo
          fimregistro

função retornaStrEmPosiçãoP(l: PNo, p: inteiro): string
início
    se l = NULO então
        retorne ""
    fimse
    se p = 1 então
        retorne l^.info
    senão
        retorne retornaStrEmPosiçãoP(l^.prox, p-1)
    fimse
fimfunção
