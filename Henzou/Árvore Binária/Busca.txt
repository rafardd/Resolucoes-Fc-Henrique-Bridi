tipos
    PNo = ponteiro para TNo
    TNo = registro
            chave : real
            info : inteiro
            esq : PNo
            dir : PNo
          fimregistro

função busca(l: PNo; ch: real): PNo
var
    temp : PNo
início
    temp <- l
    enquanto temp != NULO faça
        se ch = temp^.chave então
            retorne temp
        fimse

        se ch < temp^.chave então
            temp <- temp^.esq
        senão
            temp <- temp^.dir
        fimse
    fimenquanto
    retorne NULO
fimfunção
