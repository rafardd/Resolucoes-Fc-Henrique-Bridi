tipos
    PNo = ponteiro para TNo
    TNo = registro
            chave : inteiro
            info : real
            esq : PNo
            dir : PNo
          fimregistro

procedimento inverte(l: PNo)
var
    atual, pré, aux: PNo
início
    atual <- l

    enquanto atual <> NULO faça
        se atual^.esq = NULO então

            aux <- atual^.esq
            atual^.esq <- atual^.dir
            atual^.dir <- aux

            atual <- atual^.esq

        senão
            pré <- atual^.esq
            enquanto pré^.dir != NULO e pré^.dir != atual faça
                pré <- pré^.dir
            fimenquanto

            se pré^.dir = NULO então
                pré^.dir <- atual
                atual <- atual^.esq
            senão
                pré^.dir <- NULO

                aux <- atual^.esq
                atual^.esq <- atual^.dir
                atual^.dir <- aux

                atual <- atual^.esq
            fimse
        fimse
    fimenquanto
fimprocedimento
