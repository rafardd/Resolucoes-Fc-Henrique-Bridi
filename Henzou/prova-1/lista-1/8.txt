função converteInteiroEmStringBinária(n : inteiro) : string
var
  resto : inteiro
  str, sinal : string
início
  sinal <- ""
  str <- ""
  se n = 0 então
    str <- "0"
  senão
    se n < 0 então
      sinal <- "-"
      n <- n * -1
    fimse
    enquanto n > 0 faça
      resto <- n mod 2 
      str <- chr(resto + asc('0')) + str
      n <- n div 2
    fimenquanto
  fimse
    
  retorne (sinal + str)
fimfunção
