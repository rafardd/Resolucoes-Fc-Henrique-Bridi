função retornaModa(s: string):inteiro
var
    frequência : vetor[0..3000] de inteiro
    numAtual, i, moda, maiorModa : inteiro
    f: sequência de inteiro
início
    para i de 0 até 3000 faça
        frequência[i] <- 0
    fimpara

    se abre(f, s) então
        enquanto não fds(f) faça
            leia(f, numAtual)
            frequência[numAtual] <- frequência[numAtual] + 1
        fimenquanto
        fecha(f)

        moda <- frequência[0]
        maiorModa <- 0
        para i de 1 até 3000 faça
            se frequência[i] > moda então
                moda <- frequência[i]
                maiorModa <- i
            fimse
        fimpara

        retorne maiorModa
    senão
        retorne -9999
    fimse
fimfunção
