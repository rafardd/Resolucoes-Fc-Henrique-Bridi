função eliminaInt(s: string; x: inteiro): lógico
var
  numAtual : inteiro
  removeu : lógico
  foriginal, fnovo : sequência de inteiro
início
  se abre(foriginal, s) então
    se cria(fnovo, "temp.dat") então
      removeu <- FALSO
      enquanto não fds(foriginal) faça
        leia(foriginal, numAtual)
        se numAtual != x então
          escreva(fnovo, numAtual)
        senão
          removeu <- VERDADEIRO
        fimse
      fimenquanto
      fecha(foriginal)
      fecha(fnovo)
      apaga(s)
      renomeia("temp.dat", s)
      retorne removeu
    senão
      fecha(foriginal)
      retorne FALSO
    fimse
  senão
    retorne FALSO
  fimse
fimfunção
