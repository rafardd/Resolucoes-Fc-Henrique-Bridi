tipos
  PNo : ponteiro para TNo
  TNo : registro
          info : string
          prox : PNo
        fimregistro

função removeÚltimo(ref l: PNo): lógico
início
  se l = NULO então
    retorne FALSO
  fimse

  se l^.prox = NULO então
    desaloca(l)
    l <- NULO
    retorne VERDADEIRO
  fimse
  removeÚltimo(l^.prox)
fimfunção
