tipos
    PNo : ponteiro para TNo
    TNo : registro
            info : string
            prox : PNo
          fimregistro

função insereString(ref l: PNo; x: string): lógico
var
    novo : PNo
início
    se l = NULO ou l^.info > x então
        novo <- aloca(TNo)
        se novo = NULO então
            retorne FALSO
        fimse

        novo^.info <- x
        novo^.prox <- l
        l <- novo
        retorne VERDADEIRO
    senão
        se l^.info = x então
            retorne FALSO
        fimse
    fimse

    insereString(l^.prox, x)
fimfunção
