tipos
    PNo : ponteiro para TNo
    TNo : registro
            info : real
            prox : PNo
          fimregistro

função verificaElementosContíguos(l: PNo; X, Y : real): lógico
início
    se l != NULO então
        se l^.prox != NULO então
            se l^.info = X e l^.prox^.info = Y então
                retorne VERDADEIRO
            fimse
        fimse
        retorne retornaElementosContíguos(l^.prox, X, Y)
    senão
        retorne FALSO
    fimse
fimfunção
