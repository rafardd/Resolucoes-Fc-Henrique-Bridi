função retornaSomatório(s: string): inteiro
var
  nomeAtual : string
  numAtual, soma : inteiro
  fstring : sequência de string
  fint : sequência de inteiro
início
  se abre(fstring, s) então
    soma <- 0
    enquanto não fds(fstring) faça
      leia(fstring, nomeAtual)
      se abre(fint, nomeAtual) então
        enquanto não fds(fint) faça
          leia(fint, numAtual)
          soma <- soma + numAtual
        fimenquanto
        fecha(fint)
      senão
        fecha(fstring)
        pare
      fimse
    fimenquanto
    fecha(fstring)
    retorne soma
  senão
    pare
  fimse
fimfunção
