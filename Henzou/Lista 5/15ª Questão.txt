função retorna7Maior(s: string): real
const
  K = 7
var
  v : vetor[1..K] de real
  n : real
  i, j, z : inteiro
  inseriu : lógico
  f : sequência de real
início
  se abre(f, s) então
    i <- 1
    enquanto não fds(f) faça
      leia(f, n)
      se i = 1 então
        v[1] <- n
      senão
        j <- 1
        inseriu <- FALSO
        enquanto j <= i e não inseriu faça
          se n > v[j] então
            z <- i
            enquanto z > j faça
              v[z] <- v[z-1]
              z <- z - 1
            fimenquanto
            v[j] <- n
            inseriu <- VERDADEIRO
          fimse
          j <- j + 1
        fimenquanto
      fimse
      se i < K então
        i <- i + 1
      fimse
    fimenquanto
    fecha(f)
    se i < K então
      pare
    fimse
    retorne v[K]
  senão
    pare
  fimse
fimfunção
