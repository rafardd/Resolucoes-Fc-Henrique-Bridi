Tipo
    PNo = ponteiro para TNo
    TNo = registro
            coef: real
            exp: real
            prox: PNo
          fim registro

Função MergePolinomio(l1, l2: PNo): PNo
Var
    novo, p1, p2: PNo
    c, expo: real
Início
    Se l1 = NULO e l2 = NULO então
        Retorne NULO
    Se l2 = NULO ou (l1 ≠ NULO e l1↑.exp > l2↑.exp) então
        c ← l1↑.coef
        expo ← l1↑.exp
        p1 ← l1↑.prox
        p2 ← l2
    Senão
        Se l1 = NULO ou l2↑.exp > l1↑.exp então
            c ← l2↑.coef
            expo ← l2↑.exp
            p1 ← l1
            p2 ← l2↑.prox
        Senão
            c ← l1↑.coef + l2↑.coef
            expo ← l1↑.exp
            p1 ← l1↑.prox
            p2 ← l2↑.prox
        Fim se
    Fim se

    Se c = 0 então
        Retorne MergePolinomio(p1, p2)
    Fim se

    novo ← aloca(TNo)
    Se novo = NULO então
        Retorne NULO
    Fim se
    novo↑.coef ← c
    novo↑.exp ← expo
    novo↑.prox ← MergePolinomio(p1, p2)
    Retorne novo
Fim função
