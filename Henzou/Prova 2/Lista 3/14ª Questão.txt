função retornaMaiorValorLido(flag : real): real
var
  num, maior : real
início
  escreva("Insira um número: ")
  leia(num)
  
  se num = flag então
    retorne flag
  senão
    maior <- retornaMaiorValorLido(flag)
    se num > maior ou maior = flag então
      retorne num
    senão
      retorne maior
    fimse
  fimse
fimfunção
