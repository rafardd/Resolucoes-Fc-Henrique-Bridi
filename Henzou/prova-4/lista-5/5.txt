função retornaSomaDePosições(s: string; x: real): inteiro
var
  soma, posição : inteiro
  numAtual : real
  f : sequência de real
início
  se abre(f, s) então
    posição <- 1
    soma <- 0
    enquanto não fds(f) faça
      leia(f, numAtual)
      se numAtual = x então
        soma <- soma + posição
      fimse
      posição <- posição + 1
    fimenquanto
    fecha(f)
    retorne soma
  senão
    retorne -99999
  fimse
fimfunção
