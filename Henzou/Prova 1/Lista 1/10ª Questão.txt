função verifiExistênciaDeRaízReal(ref r1, r2 : real; a, b, c : real) : lógico

  delta : real
início
  se a = 0 então
     pare
  fimse

  delta <- b*b  -  4*a*c
  se delta < 0 então
    retorne FALSO
  senão
    r1 <- (-b  -  raizq(delta)) / (2*a)
    r2 <- (-b  +  raizq(delta)) / (2*a)
