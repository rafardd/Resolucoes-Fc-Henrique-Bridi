função verificaPalíndrome(s : string) : lógico
var
  igual : lógico
  i : inteiro
início
  se tamanho(s) = 0 então
    retorne FALSO
  fimse

  para i de 1 até tamanho(s) faça
    se 'A' <= s[i] e s[i] <= 'Z' então
      s[i] <- chr(asc(s[i]) + (asc('a') - asc('A')))
    fimse
  fimpara
  
  igual <- VERDADEIRO
  para i de 1 até tamanho(s) div 2 faça
    igual <- igual e (s[i] = s[tamanho(s) - i + 1])
  fimpara

  retorne igual
fimfunção
