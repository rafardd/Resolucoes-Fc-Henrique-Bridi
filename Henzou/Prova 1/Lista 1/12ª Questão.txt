função retornaStringComCaracteresSequenciais(car : caracter; num : inteiro) : string
var
  limite : inteiro
  resultado : string
início
  resultado <- ""
  limite <- asc(car) + num

  se num < 0 então
    enquanto asc(car) >= 32 e asc(car) >= limite faça
      resultado <- resultado + car
      car <- chr(asc(car) - 1)
    fimenquanto
  senão	
    enquanto asc(car) <= 126 e asc(car) <= limite faça
      resultado <- resultado + car
      car <- chr(asc(car) + 1)
      fimenquanto
  fimse

  retorne resultado
fimfunção
