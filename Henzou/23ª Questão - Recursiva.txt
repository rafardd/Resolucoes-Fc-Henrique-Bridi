tipos
    PNo : ponteiro para TNo
    TNo : registro
            info : string
            prox : PNo
          fimregistro

função removeElementos(ref l: PNo; x: string): lógico
var
    temp : PNo
    removeu : lógico
início
    se l = NULO então
        retorne FALSO
    fimse

    se l^.info >= x então
        temp <- l
        l <- temp^.prox
        desaloca(temp)
        removeu <- VERDADEIRO
        retorne removeu ou removeElementos(l, x)
    fimse
    retorne removeElementos(l^.prox, x)
fimfunção
