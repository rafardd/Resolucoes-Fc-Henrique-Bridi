tipos 
  PNo = ponteiro para TNo
  TNo = registro
          info : string
          prox : PNo
        fimregistro

função removeValor(ref l: PNo; x: string): lógico
var
  temp : PNo
início
  se l = NULO então
    retorne FALSO
  fimse

  se l^.info = x então
    temp <- l
    l <- temp^.prox
    desaloca(temp)
    retorne VERDADEIRO
  fimse

  retorne removeValor(l^.prox, x)
fimfunção
