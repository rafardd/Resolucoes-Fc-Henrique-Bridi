função procuraValoresRepetidos(l: PNo; X: real): inteiro
var
    total : inteiro
início
    total <- 0
    enquanto l != NULO faça
        se l^.info = X então
            total <- total + 1
        fimse
        l <- l^.prox
    fimenquanto
    retorna total
fimfunção
