função removePrimeiro(ref l: PNo): lógico
var
  temp : PNo
início
  se l = NULO então
    retorne FALSO
  fimse
  temp <- l
  l <- l^.prox
  desaloca(temp)
  retorne VERDADEIRO
fimfunção
