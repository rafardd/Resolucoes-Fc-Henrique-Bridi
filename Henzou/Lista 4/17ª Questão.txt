tipos
    PNo = ponteiro para TNo
    TNo = registro
            info : string
            prox : PNo
          fimregistro

função insereString(ref l: PNo; x: string): lógico
var
    novo, temp, ant : PNo
início
    temp <- l
    ant <- NULO
    enquanto temp != NULO e temp^.info < x faça
        ant <- temp
        temp <- temp^.prox
    fimenquanto

    se l != NULO e temp^.info = x então
        retorne FALSO
    fimse

    novo <- aloca(TNo)
    se novo = NULO então
        retorne FALSO
    fimse
    novo^.info <- x
    novo^.prox <- temp

    se ant = NULO então
        l <- novo
    senão
        ant^.prox <- novo
    fimse
    retorne VERDADEIRO
fimfunção
