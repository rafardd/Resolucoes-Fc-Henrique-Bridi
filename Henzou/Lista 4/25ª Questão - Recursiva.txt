tipos
    PNo = ponteiro para TNo
    TNo = registro
            info : real
            prox : PNo
          fimregistro

procedimento passaÚltimoParaInício(ref l: PNo)
var
    temp : PNo
início
    se l != NULO e l^.prox != NULO então
        se l^.prox^.prox = NULO então
            temp <- l^.prox
            l^.prox <- temp^.prox
            temp^.prox <- l
            l <- temp
        senão
            passaÚltimoParaInício(l^.prox)
            temp <- l^.prox
            l^.prox <- temp^.prox
            temp^.prox <- l
            l <- temp
        fimse
    fimse
fimprocedimento
