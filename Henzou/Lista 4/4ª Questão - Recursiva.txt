tipos
    PNo : ponteiro para TNo
    TNo : registro
            info : real
            prox : PNo
          fimregistro

função procuraValoresRepetidos(l: PNo; X: real): inteiro
início
    se l = NULO então
        retorne 0
    senão
        se l^info = X então
            retorne 1 + procuraValoresRepetidos(l^.prox, X)
        senão
            retorne procuraValoresRepetidos(l^.prox, X)
        fimse
    fimse
fimfunção
