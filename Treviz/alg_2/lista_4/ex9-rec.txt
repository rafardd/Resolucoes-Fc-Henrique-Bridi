TIPOS
    PNo = PONTEIRO PARA TNo
    TNo =
        REGISTRO
            elemento : real
            prox : PNo
        FIM REGISTRO

FUNÇÃO busca_x_y_juntos(l:PNo ; x,y:real):lógico
INICIO
    SE l = NULO ENTAO
        RETORNE falso
    FIM SE
    SE l^.prox = NULO ENTAO
        RETORNE falso
    FIM SE
    SE x = l^.elemento E y = l^.prox^.elemento ENTAO
        RETORNE verdadeiro
    FIM SE
    RETORNE busca_x_y_juntos(l^.prox,x,y)
FIM FUNÇÃO
