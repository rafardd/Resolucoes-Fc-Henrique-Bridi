TIPO
    PNo = PONTEIRO PARA TNo
    TNo = 
        REGISTRO
            info : real
            prox : PNo
        FIM REGISTRO

FUNÇÃO remove_e_conta(REF l:PNo):inteiro
VAR
    ant, lixo: PNo
    qtd : inteiro
INICIO
    qtd <- 0
    ant <- NULO
    lixo <- l
    ENQUANTO lixo != NULO ENTAO
        SE ant!= NULO E ant^.info = lixo^.info ENTAO
            ant^.prox <- lixo^.prox
            desaloca(lixo)
            lixo <- ant^.prox
            qtd <- qtd + 1
        SENAO
            ant <- lixo
            lixo <- lixo^.prox
        FIM SE
    FIM ENQUANTO
    RETORNE qtd
FIM FUNÇÃO
