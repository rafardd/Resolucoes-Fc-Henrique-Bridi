TIPOS
    PNo = PONTEIRO PARA TNo
    TNo = 
        REGISTRO
            info : real
            prox : PNo
        FIM REGISTRO

PROCEDIMENTO os_ultimos_serao_os_primeiros(REF l:PNo)
VAR
    aux : PNo
INICIO
    SE l != NULO E l^.prox != NULO ENTAO
        os_ultimos_serao_os_primeiros(l^.prox)
        aux <- l^.prox
        l^.prox <- aux^.prox
        aux^.prox <- l
        l <- aux
    FIM SE
FIM PROCEDIMENTO
