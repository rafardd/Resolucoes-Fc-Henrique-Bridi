TIPOS
    PNo = PONTEIRO PARA TNo
    TNo =
        REGISTRO
            info : string
            prox : PNo
        FIM REGISTRO
    
FUNÇÃO remove_x_e_maior(REF l:PNo ; x:string):lógico
VAR
    lixo : PNo
INICIO
    SE l = NULO ENTAO
        RETORNE falso
    FIM SE
    SE l^.info >= x ENTAO
        lixo <- l
        l <- lixo^.prox
        desaloca(lixo)
        lixo <- NULO
        RETORNE remove_x_e_maior(l, x) OU VERDADEIRO
    FIM SE
    RETORNE remove_x_e_maior(l^.prox, x)
FIM FUNÇÃO
