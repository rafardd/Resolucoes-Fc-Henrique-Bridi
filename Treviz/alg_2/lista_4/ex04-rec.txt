funcao count(l:PNo ; x:real):inteiro
inicio
    se l = nulo entao
        retorne 0
    senao
        se l^.info = x entao
            retorne count(l^.prox ; x) + 1
        senao
            retorne count(l^.prox ; x)
        fim se
    fimse
fim funcao
