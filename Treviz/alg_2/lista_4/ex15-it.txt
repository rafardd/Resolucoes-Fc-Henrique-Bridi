TIPOS
    PNo = PONTEIRO PARA TNo
    TNo = 
        REGISTRO
            chave : inteiro
            info : string
            prox : PNo
        FIM REGISTRO

FUNÇÃO insere_no_final(REF l:PNo, ch: inteiro, info:string):lógica
VAR
    aux, ant, novo: PNo
INICIO
    aux <- l
    ant <- NULO
    ENQUANTO aux != NULO FAÇA
        ant <- aux
        aux <- aux^.prox
    FIM ENQUANTO
    novo <- aloca(TNo)
    SE novo = NULO ENTAO
        RETORNE falso
    FIM SE
    novo^.chave <- ch
    novo^.info <- inf
    novo^.prox <- NULO
    SE ant = NULO ENTAO
        l <- novo
    SENÃO
        ant^.prox <- novo
    FIM SE
    RETORNE verdadeiro
FIM FUNÇÃO
