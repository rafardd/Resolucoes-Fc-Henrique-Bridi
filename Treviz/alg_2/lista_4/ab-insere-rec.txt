TIPOS
    PNo = PONTEIRO PARA TNo
    TNo = 
        REGISTRO
            chave : inteiro
            info : string
            esq : PNo
            dir : PNo
        FIM REGISTRO

FUNÇÃO insere(REF l:PNo ; ch:inteiro ; inf:string):lógico
INICIO
    SE l = NULO ENTAO
        l <- aloca(PNo)
        SE l = NULO ENTAO
            RETORNE falso
        FIM SE
        l^.chave <- ch
        l^.info <- inf
        l^.esq <- NULO
        l ^.dir <- NULO
        RETORNE verdadeiro
    FIM SE
    SE l^.chave = ch ENTAO
        RETORNE falso
    FIM SE
    SE ch > l^.chave ENTAO
        RETORNE insere(l^.dir, ch, inf)
    SENAO
        RETORNE insere(l^.esq, ch, inf)
    FIM SE
FIM FUNÇÃO
