TIPOS
    PNo = PONTEIRO PARA TNo
    TNo = 
        REGISTRO
            chave : inteiro
            prox : PNo
        FIM REGISTRO

FUNÇÃO remove_ultimo(REF l:PNo):lógico
VAR
    ant, lixo : PNo
INICIO
    SE l != NULO ENTAO
        ant <- NULO
        lixo <- l
        ENQUANTO lixo^.prox != NULO FAÇA
            ant <- lixo
            lixo <- lixo^.prox
        FIM ENQUANTO
        SE lixo = l ENTAO
            l <- lixo^.prox
        SENAO
            ant^.prox <- lixo^.prox
        FIM SE
        desaloca(lixo)
        lixo <- NULO
        RETORNE verdadeiro
    FIM SE
    RETORNE falso
FIM FUNÇÃO
