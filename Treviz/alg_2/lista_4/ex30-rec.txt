TIPOS
    PNo = PONTEIRO PARA TNo
    TNo =
        REGISTRO
            coef : real
            exp : real
            prox : PNo
        FIM REGISTRO

FUNCAO soma_polinomio(p1,p2: PNo):PNo
VAR
    head : PNo

    FUNÇÃO soma(REF p1,p2,p3:PNo)
    VAR
        coef : real

        PROCEDIMENTO cria(REF pp:PNo ; c,e:real)
        INICIO
            pp <- aloca(TNo)
            SE pp = NULO ENTAO
                PARE
            FIM SE
            pp^.coef <- c
            pp^.exp <- e
            pp^.prox <- NULO
        FIM PROCEDIMENTO

    INICIO
        SE p1 = NULO E p2 = NULO ENTAO
            RETORNE head
        SENAO
            SE p1 = NULO ENTAO
                cria(p3, p2^.coef, p2^.exp)
                RETORNE soma(p1, p2^.prox, p3^.prox)
            FIM SE
            SE p2 = NULO ENTAO
                cria(p3, p1^.coef, p1^.exp)
                RETORNE soma(p1^.prox, p2, p3^.prox)
            FIM SE
            SE p1^.exp = p2^.exp ENTAO
                coef <- p1^.coef + p2^.coef
                SE coef != 0 ENTAO
                    cria(p3, coef, p1^.exp)
                FIM SE
                RETORNE soma(p1^.prox, p2^.prox, p3^.prox)
            SENAO
                SE p1^.exp > p2^.exp ENTAO
                    cria(p3, p1^.coef, p1^.exp)
                    RETORNE soma(p1^.prox, p2, p3^.prox)
                SENAO
                    cria(p3, p2^.coef, p2^.exp)
                    RETORNE soma(p1, p2^.prox, p3^.prox)
                FIM SE
            FIM SE
        FIM SE
    FIM FUNÇÃO

INICIO
    head <- NULO
    RETORNE soma(p1, p2, head)
FIM FUNÇÃO
