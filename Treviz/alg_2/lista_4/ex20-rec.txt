TIPOS
    PNo = PONTEIRO PARA TNo
    TNo = 
        REGISTRO
            chave : inteiro
            prox : PNo
        FIM REGISTRO

FUNÇÃO remove_ultimo(REF l:PNo):lógico
VAR
    lixo : PNo
INICIO
    SE l = NULO ENTAO
        RETORNE falso
    FIM SE
    SE l^.prox = NULO ENTAO
        lixo <- l
        l <- lixo^.prox
        desaloca(lixo)
        lixo <- NULO
        RETORNE verdadeiro
    FIM SE
    RETORNE remove_ultimo(l^.prox)
FIM FUNÇÃO
