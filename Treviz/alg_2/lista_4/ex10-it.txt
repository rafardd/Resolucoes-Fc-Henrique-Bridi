TIPOS
    PNo = PONTEIRO PARA TNo
    TNo = 
        REGISTRO    
            elemento : real
            prox : PNo
        FIM REGISTRO

FUNÇÃO busca_x_y(l:PNo ; x,y:real):lógico
INICIO
    ENQUANTO l != NULO E x != l^.elemento FAÇA
        l <- l^.prox
    FIM ENQUANTO
    SE l = NULO ENTAO
        RETORNE falso
    FIMSE
    l <- l^.prox
    ENQUANTO l != NULO E y != l^.elemento FAÇA
        l <- l^.prox
    FIM ENQUANTO
    SE l = NULO ENTAO
        RETORNE falso
    FIMSE
    RETORNE verdadeiro
FIM FUNÇÃO
