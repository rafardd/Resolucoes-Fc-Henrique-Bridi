    TNo = 
        REGISTRO    
            elemento : real
            prox : PNo
        FIM REGISTRO

FUNÇÃO busca_x_y(l:PNo ; x,y:real):lógico
VAR
    FUNÇÃO busca_e(p:PNo ; e:real):lógico
    INICIO
        SE p = NULO ENTAO
            RETORNE falso
        FIM SE
        SE e = p^.elemento ENTAO
            RETORNE verdadeiro
        FIM SE
        RETORNE busca_e(p^.prox, e)
    FIM FUNÇÃO

INICIO
    SE l = NULO ENTAO
        RETORNE falso
    FIM SE
    SE x = l^.elemento E busca_elemento(l^.prox, y) ENTAO
        RETORNE verdadeiro
    FIM SE
    RETORNE busca_x_y(l^.prox, x, y)
FIM FUNÇÃO
