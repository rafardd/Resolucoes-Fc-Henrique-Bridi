TIPOS
    PNo = PONTEIRO PARA TNo
    TNo = 
        REGISTRO
            info : real
            prox : PNo
        FIM REGISTRO
    
FUNÇÃO soma_elementos(REF l:PNo ; p:inteiro):lógico
VAR
    aux, lixo: PNo
INICIO
    SE l != NULO E l^.prox != NULO ENTAO
        aux <- l
        ENQUANTO aux^.prox != NULO E p > 1 FAÇA
            aux <- aux^.prox
            p < p - 1
        FIM ENQUANTO
        SE p = 1 ENTAO
            lixo <- aux^.prox
            aux^.info <- aux^.info + lixo^.info
            aux^.prox <- lixo^.prox
            desaloca(lixo)
            RETORNE verdadeiro
        FIM SE
    FIM SE
    RETORNE falso
FIM FUNÇÃO
