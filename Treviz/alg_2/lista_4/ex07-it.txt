TIPOS
    PNo = PONTEIRO PARA TNo
    TNo = 
        REGISTRO
            chave : inteiro
            info : inteiro
            prox : PNo
        FIM REGISTRO

FUNCAO media(l:PNo):real
VAR
    soma, n: inteiro
INICIO
    soma <- 0
    n <- 0
    ENQUANTO l != nulo FACA
        soma = soma + l^.info
        l <- l^.prox
        n <- n + 1
    FIM ENQUANTO
    SE n = 0 ENTAO 
        RETORNA 0
    FIM SE
    RETORNA soma / n
FIM FUNCAO
