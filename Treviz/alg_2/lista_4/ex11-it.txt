TIPOS
    PNo = PONTEIRO PARA TNo
    TNo = 
        REGISTRO
            elemento : real
            prox : PNo
        FIM REGISTRO

FUNÇÃO busca_x(l:PNo ; x:real):lógico
INICIO
    ENQUANTO l != NULO E x < l^.elemento FAÇA
        l <- l^.prox
    FIM ENQUANTO
    SE l != NULO E x = l^.elemento ENTÃO
        RETORNE verdadeiro
    FIM SE
    RETORNE falso
FIM FUNÇÃO
