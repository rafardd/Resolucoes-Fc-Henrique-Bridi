TIPOS
    PNo = PONTEIRO PARA TNo
    TNo =
        REGISTRO
            coef : real
            exp : real
            prox : PNo
        FIM REGISTRO

FUNCAO soma_polinomio(p1,p2: PNo):PNo
VAR
    novo : PNo
    coef : real
INICIO
    SE p1 = NULO E p2 = NULO ENTAO
            RETORNE NULO
    SENAO
        SE p1 = NULO ENTAO
            novo <- aloca(TNo)
            SE novo = NULO ENTAO
                RETORNE NULO
            FIM SE
            novo^.coef <- p2^.coef
            novo^.exp <- p2^.exp
            novo^.prox <- soma_polinomio(p1, p2^.prox)
            RETORNE novo
        FIM SE
        SE p2 = NULO ENTAO
            novo <- aloca(TNo)
            SE novo = NULO ENTAO
                RETORNE NULO
            FIM SE
            novo^.coef <- p1^.coef
            novo^.exp <- p1^.exp
            novo^.prox <- soma_polinomio(p1^.prox, p2)
            RETORNE novo            
        FIM SE
        SE p1^.exp = p2^.exp ENTAO
            coef <- p1^.coef + p2^.coef
            SE coef = 0 ENTAO
                RETORNE soma_polinomio(p1^.prox, p2^.prox)
            SENAO
                novo <- aloca(TNo)
                SE novo = NULO ENTAO
                    RETORNE NULO
                FIM SE
                novo^.coef <- p1^.coef
                novo^.exp <- p1^.exp
                novo^.prox <- soma_polinomio(p1^.prox, p2^.prox)
                RETORNE novo
            FIM SE     
        SENAO
            SE p1^.exp > p2^.exp ENTAO
                novo <- aloca(TNo)
                SE novo = NULO ENTAO
                    RETORNE NULO
                FIM SE
                novo^.coef <- p1^.coef
                novo^.exp <- p1^.exp
                novo^.prox <- soma_polinomio(p1^.prox, p2)
                RETORNE novo
            SENAO
                novo <- aloca(TNo)
                SE novo = NULO ENTAO
                    RETORNE NULO
                FIM SE
                novo^.coef <- p2^.coef
                novo^.exp <- p2^.exp
                novo^.prox <- soma_polinomio(p1, p2^.prox)
                RETORNE novo
            FIM SE
        FIM SE
    FIM SE
FIM FUNÇÃO
