TIPOS
    PNo = PONTEIRO PARA TNo
    TNo = 
        REGISTRO
            info : real
            prox : PNo
        FIM REGISTRO
    
FUNÇÃO soma_elementos(REF l:PNo ; p:inteiro):lógico
VAR
    lixo : PNo
INICIO
    SE l = NULO OU l^.prox = NULO ENTAO
        RETORNE falso
    FIM SE
    SE p = 1 ENTAO
        lixo <- l^.prox
        l^.info <- l^.info + lixo^.info
        l^.prox <- lixo^.prox
        desaloca(lixo)
        RETORNE verdadeiro
    FIM SE
    RETORNE soma_elementos(l^.prox, p-1)
FIM FUNÇÃO
