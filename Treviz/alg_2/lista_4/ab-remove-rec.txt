TIPOS
    PNo = PONTEIRO PARA TNo
    TNo = 
        REGISTRO
            chave : inteiro
            info : string
            esq : PNo
            dir : PNo
        FIM REGISTRO

FUNÇÃO remove(REF l:PNo; ch:inteiro):lógico
VAR
    lixo : PNo

    FUNÇÃO procura_maior(l:PNo):PNo
    INICIO
        SE l^.dir = NULO ENTAO
            RETORNE l
        FIM SE
        RETORNE procura_maior(l^.dir)
    FIM FUNÇÃO
INICIO
    SE l = NULO ENTAO           // NÃO ENCONTROU A CHAVE
        RETORNE falso
    FIM SE
    SE ch > l^.chave ENTAO      // PROCURA CHAVE
        RETORNE insere(l^.dir, ch, inf)
    FIM SE
    SE ch < l^.chave ENTAO      // PROCURA CHAVE
        RETORNE insere(l^.esq, ch, inf)
    FIM SE
    SE l^.esq = l^.dir ENTAO        // ENCONTROU A CHAVE. SEM FILHOS
        lixo <- l
        l <- NULO      
        desaloca(lixo)
        RETORNE verdadeiro
    FIM SE
    SE l^.esq = NULO ou l^.dir = NULO ENTAO     // APENAS UM FILHO
        lixo <- l
        SE l^.esq != NULO ENTAO
            l <- l^.esq
        SENAO
            l <- l^.dir
        FIM SE
        desaloca(lixo)
        RETORNE verdadeiro
    FIM SE
    lixo <- procura_maior(l^.esq)       // DOIS FILHOS. ESCOLHI O DA ESQUERDA
    l^.chave <- lixo^.chave
    l^.info <- lixo^.info
    RETORNE remove(l^.esq, lixo^.chave)
FIM FUNÇÃO
