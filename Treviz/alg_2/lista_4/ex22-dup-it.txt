TIPOS
    PNo = PONTEIRO PARA TNo
    TNo =
        REGISTRO
            info : string
            prox : PNo
            prev : PNo
        FIM REGISTRO

FUNÇÃO remove_x(REF l:PNo ; x:string):lógico
VAR
    lixo, ant : PNo
INICIO
    SE l != NULO ENTAO
        ant <- NULO
        lixo <- l
        ENQUANTO l^.prox != NULO E lixo^.prox != x FAÇA
            ant <- lixo
            lixo <- lixo^.prox
        FIM ENQUANTO
        SE l^.info = x ENTAO
            SE ant = NULO ENTAO
                l <- lixo^.prox
            SENAO
                ant^.prox <- lixo^.prox
            FIM SE
            SE lixo^.prox != NULO ENTAO
                lixo^.prox^.prev <- ant
            FIM SE
            desaloca(lixo)
            RETORNE verdadeiro
        FIM SE
    FIM SE
    RETORNE falso
FIM FUNÇÃO
