TIPOS
    PNo = PONTEIRO PARA TNo
    TNo =
        REGISTRO
            info : string
            prox : PNo
            prev : PNo
        FIM REGISTRO

FUNÇÃO remove_x(REF l:PNo ; x:string):lógico
VAR
    lixo
INICIO
    SE l = NULO ENTAO
        RETORNE falso
    FIM SE
    SE l^.info = x ENTAO
        lixo <- l
        l <- l^.prox
        SE lixo^.prox != NULO ENTAO
            lixo^.prox^.prev <- lixo^.prev
        FIM SE
        desaloca(lixo)
        RETORNE verdadeiro
    FIM SE
    RETORNE remove_x(l^.prox, x)
FIM FUNÇÃO
