TIPOS
    PNo = PONTEIRO PARA TNo
    TNo = 
        REGISTRO
            chave : inteiro
            prox : PNo
        FIM REGISTRO
    
FUNÇÃO remove_em_p(REF l:PNo ; p:inteiro)lógico
VAR
    ant, lixo : PNo
INICIO
    SE l != NULO ENTAO
        ant <- NULO
        lixo <- l
        ENQUANTO lixo^.prox != NULO E p > 1 FAÇA
            ant <- lixo
            lixo <- lixo^.prox
            p <- p - 1
        FIM ENQUANTO
        SE p = 1 ENTAO
            SE lixo = l ENTAO
                l <- lixo^.prox
            SENAO
                ant^.prox <- lixo^.prox
            FIM SE
            desaloca(lixo)
            lixo <- NULO
            RETORNE verdadeiro
        FIM SE
    FIM SE
    RETORNE falso
FIM FUNÇÃO
