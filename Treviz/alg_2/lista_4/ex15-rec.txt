TIPOS
    PNo = PONTEIRO PARA TNo
    TNo = 
        REGISTRO
            chave : inteiro
            info : string
            prox : PNo
        FIM REGISTRO

FUNÇÃO insere_no_final(REF l:PNo, ch: inteiro, info:string):lógica
VAR
    novo: TNo
INICIO
    SE l = NULO ENTAO
        novo <- aloca(TNo)
        SE novo = NULO ENTAO
            RETORNE falso
        FIM SE
        novo^.chave <- ch
        novo^.info <- inf
        novo^.prox <- NULO
        l <- novo
        RETORNE verdadeiro
    SENÃO
        RETORNE insere_no_final(l^.prox, ch, inf)
    FIM SE
FIM FUNÇÃO
