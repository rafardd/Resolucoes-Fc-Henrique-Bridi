TIPOS
    PNo = PONTEIRO PARA TNo
    TNO = 
        REGISTRO
            info : inteiro
            prox : PNo
        FIM REGISTRO
    
PROCEDIMENTO filtra_paridade(l:PNo ; REF par, impar:PNo)
INICIO
    SE l != NULO ENTAO
        SE l^.info MOD 2 = 0 ENTAO
            par <- aloca(TNo)
            SE par = NULO ENTAO
                NULO
            FIM SE
            par^.info <- l^.info
            par^.prox <- NULO
            RETORNE filtra_paridade(l^.prox, par^.prox, impar)
        SENAO
            impar <- aloca(TNo)
            SE impar = NULO ENTAO
                NULO
            FIM SE
            impar^.info <- l^.info
            impar^.prox <- NULO
            RETORNE filtra_paridade(l^.prox, par, impar^.prox)
        FIM SE
    FIM SE
FIM FUNÇÃO
