TIPOS
    PNo = PONTEIRO PARA TNo
    TNo = 
        REGISTRO
            chave : inteiro
            prox : PNo
        FIM REGISTRO
    
FUNÇÃO remove_em_p(REF l:PNo ; p:inteiro)lógico
VAR
    lixo : PNo
INICIO
    SE p <= 0 OU l = NULO ENTAO
        RETORNE falso
    FIM SE
    SE p = 1 ENTAO
        lixo <- l
        l <- lixo^.prox
        desaloca(lixo)
        lixo <- NULO
        RETORNE verdadeiro
    FIM SE
    RETORNE remove_em_p(l^.prox, p - 1)
FIM SE
