const
    K = 1000
funcao verifica_mediana(s:string):inteiro
var
    v : vetor[0..K] de inteiro
    f : sequencia de inteiro
    xf, n, i, aux: inteiro
    mediana: real
    trocou : logico
inicio
    se abrir(f,s) entao
        n <- 0
        enquanto NAO fds(f) faca
            leia(f, xf)
            v[n] <- xf
            n <- n +1
        fim enquanto
        se n = 0 entao
            mediana <- 0
        senao
            trocou <- VERDADEIRO
            enquanto trocou faca
                trocou <- falso
                para i de 0 ate n-2 faca
                    se v[i] > v[i+1] entao
                        aux <- v[i]
                        v[i] <- v[i+1]
                        v[i+1] <- aux
                        trocou <- VERDADEIRO
                    fim se
                fim para
            fim enquanto
            se n mod 2 = 0 entao
                mediana <- (v[(n mod 2) - 1] + v[n mod 2])/2
            senao
                mediana <- v[n mod 2]
            fim se
        fim se
        fecha(f)
    senao
        mediana <- 40404
    fim se
    retorne mediana
fim funcao
