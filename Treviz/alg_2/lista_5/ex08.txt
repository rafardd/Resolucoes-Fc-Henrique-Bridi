funcao inverteP(s:string; p:inteiro):logico
var
    xp, xp1, x: real
    f, t: sequencia de real
    tam, i : inteiro
inicio
    se p > 0 entao
        se abre(f, s) entao
            tam <- 0
            enquanto NAO fds(f) faca
                leia(f, x)
                tam <- tam + 1
            fim enquanto
            fecha(f)
            se tam >= p+1 entao
                se abre(t, s) entao
                    abre(f)
                    para i de 1 ate p-1 faca
                        leia(f, x)
                        leia(t, x)
                    fim para
                    leia(t, xp)
                    leia(t, xp1)
                    escreva(f, xp1)
                    escreva(f, xp)
                    fecha(f)
                    fecha(t)
                    retorne verdadeiro
                fim se
            fim se
        fim se
    fim se
    retorne falso
fim funcao
