funcao inverteP(s:string; p:inteiro):logico
var
    xp, xp1, x: real
    f, t: sequencia de real
    tam, i : inteiro
inicio
    se p > 0 entao
        se abre(f, s) entao
            se cria(t, "temp") entao
                tam <- 0
                enquanto NAO fds(f) faca
                    leia(f, x)
                    tam <- tam + 1
                    se tam = p entao
                        xp <- x
                    fim se
                    se tam = p+1 entao
                        xp1 <- x
                    fim se
                fim enquanto
                fecha(f)
                se abre(f, s) entao
                    se tam >= p+1 entao
                        i <- 0
                        enquanto NAO fds(f) faca
                            leia(f, x)
                            i <- i + 1
                            se i = p entao
                                escreva(t, xp1)
                                escreva(t, xp)
                                i <- i + 1
                            senao
                                escreva(t, x)
                            fim se
                        fim enquanto
    fim se
    retorne falso
fim funcao
