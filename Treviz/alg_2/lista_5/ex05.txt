funcao somaDePosicao(s:string;r:real):inteiro
var
    x : real
    soma, i: inteiro
    f : sequencia de real
inicio
    se abre(f, s) entao
        i <- 0
        soma <- 0
        enquanto NAO fds(f) faca
            leia(f, x)
            i <- i + 1
            se x = r entao
                soma <- soma + i
            fim se
        fim enquanto
        fecha(f)
        retorne soma
    senao
        escreva("O arquivo ", s, "nÃ£o pode ser aberto")
        retorne -40404
    fim se
fim funcao
