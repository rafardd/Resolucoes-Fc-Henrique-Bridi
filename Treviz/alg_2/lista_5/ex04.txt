procedimento escreveMetade(s:string)
var
    i , tam : inteiro
    x : real
    f : sequencia de real
inicio
    tam <- 0
    se abre(f, s) entao
        enquanto NAO fds(f) faca
            leia(f, x)
            tam <- tam + 1
        fim enquanto
        fecha(f)
        se tam > 0 entao
            abre(f, s)
            para i de 1 ate tam faca
                leia(f, x)
                se i > tam/2 entao
                    escreva(x)
                fim se
            fim para
            fecha(f)
        fim se
    fim se
fim procedimento
