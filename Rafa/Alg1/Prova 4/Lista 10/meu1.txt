algoritmo verifica_se_A_B_sao_iguais
var
  A,B:vetor[1..100] de vetor[1..100] de inteiros
  i,j,n:inteiro
  iguais:logico 
inicio
  escreva("insira o valor de n: ")
  leia(n)
  enquanto n < 1 ou n > 100 faca
    escreva("deve estar entre 1 e 100")
    leia(n)
  fimenquanto
  para i de 1 de ate n faca
    para j de 1 ate n faca
      leia(A[i][j])
      enquanto A[i][j] < 6 ou A[i][j] > 10 faca
        escreva("deve estar entre 6 e 10")
        leia(A[i][j])
      fimenquanto
    fimpara
  fimpara
  para i de 1 de ate n faca
    para j de 1 ate n faca
      leia(B[i][j])
      enquanto B[i][j] < 6 ou B[i][j] > 10 faca
        escreva("deve estar entre 6 e 10")
        leia(B[i][j])
      fimenquanto
    fimpara
  fimpara
  iguais <- VERDADEIRO
  para i de 1 ate n faca
    para j de 1 ate n faca
      iguais <- iguais e A[i][j] = B[i][j]
    fimpara
  fimpara
  se iguais entao
    escreva("sao iguais")
  senao 
    escreva("nao sao iguais")
  fimse
fimalgoritmo
