algoritmo multiplicacao_de_matrizes
const
  MAX= 4
  MIN= -5
var
  R,S,Q:vetor[1..3] de vetor[1..3] de real
  i,j,k,nR,mR,nS,mS:inteiro
inicio
  escreva("insira o numero de linhas de R: ")
  leia(nR)
  enquanto nR < 1 ou nR > 3 faca
    escreva("deve estar entre 1 e 3")
    leia(nR)
  fimenquanto
  escreva("insira o numero de colunas de R: ")
  leia(mR)
  enquanto mR < 1 ou mR > 3 faca
    escreva("deve estar entre 1 e 3")
    leia(mR)
  fimenquanto
  escreva("insira o numero de linhas de S: ")
  leia(nS)
  enquanto nS < 1 ou nS > 3 faca
    escreva("deve estar entre 1 e 3")
    leia(nS)
  fimenquanto
  escreva("insira o numero de colunas de S: ")
  leia(mS)
  enquanto mS < 1 ou mS > 3 faca
    escreva("deve estar entre 1 e 3")
    leia(mS)
  fimenquanto
  se mR = nS entao
    para i de 1 ate mR faca
      para j de 1 ate nR faca
        R[j][i] <- trunca(((MAX - MIN)*1000 + 1)*aleatorio()) mod ((MAX - MIN)*1000 + 1) + MIN
      fimpara
    fimpara
    para i de nS faca
      para j de 1 ate mS faca
        S[i][j] <- trunca(((MAX - MIN)*1000 + 1)*aleatorio() mod ((MAX - MIN) * 1000 + 1) + MIN
      fimpara
    fimpara
    para i de 1 ate nR faca
      para j de 1 ate mS faca
        Q[i][j] <- 0 
        para k de 1 ate mR faca
         Q[i][j] <- Q[i][j]+ R[i][k] * S[k][j]
        fimpara
      fimpara
    fimpara
    para i de 1 ate nR faca
      para j de 1 ate mR faca
        escreva(R[i][j])
      fimpara
    fimpara
    para i de 1 ate nS faca
      para j de 1 ate mS faca
        escreva(S[i][j])
      fimpara
    fimpara
    para i de 1 ate nR faca
      para j de 1 ate mS faca
        escreva(Q[i][j])
      fimpara
    fimpara
  senao
    escreva("a operacao nao e possivel")
  fimse
fimalgoritmo
