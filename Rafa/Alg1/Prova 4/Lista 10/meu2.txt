algoritmo sorteia_M_e_diz_se_identidade
var
  M:vetor[1..100] de vetor[1..100] de real
  n,i,j:inteiro
  identidade:logico
inicio
  escreva("insira o valor de n: ")
  leia(n)
  para i de 1 ate n faca
    para j de 1 ate n faca
      M[i][j] <- trunca((1000000 + 1) * aleatorio()) mod (100000+1) / 1000000 
    fim para
  fimpara
  identidade <- verdadeiro
  i <- 1 
  j <- 1 
  enquanto i <= n e identidade faca
    j <- 1
    enquanto j <= n e identidade faca
      se i = j entao
        identidade <- identidade e M[i][j] = 1
      senao 
        identidade <- identidade e M[i][j] = 0
      fimse
      j <- j + 1
    fimenquanto
    i <- i + 1 
  fimenquanto
  se identidade entao
    escreva("é a matriz identidade")
  senao
    escreva("nao é a matriz identidade")
  fimse
fimalgoritmo
