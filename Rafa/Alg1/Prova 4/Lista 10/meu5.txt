algoritmo troca_linha_da_matriz
var
  F:vetor[1..100] de vetor[1.100] de caracter
  m,n,i,j,p,q:inteiro
  temporario:caracter
inicio
  escreva("insira o valor de m e n: ")
  leia(m,n)
  enquanto m < 1 ou n < 1 ou m >100 ou n > 100 faca
    escreva("os valores devem estar entre 1 e 100")
    leia(m,n)
  fimenquanto
  para i de 1 ate m faca
    para j de 1 ate n faca
      leia(F[i][j])
    fimpara
  fimpara
  escreva("insira o numero das duas linhas a serem trocadas: ")
  leia(p,q)
  para j de 1 ate n faca
    temporario <- F[q][j]
    F[q][j] <- F[p][j]
    F[p][j] <- temporario
  fimpara
  para i de 1 ate m faca
    para j de 1 ate n faca
      escreva(F[i][j])
    fimpara
  fimpara
fimalgoritmo
