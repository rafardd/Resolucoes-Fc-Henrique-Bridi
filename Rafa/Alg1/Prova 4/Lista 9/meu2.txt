algoritmo troca_mais_distante_selection_sort
var
  v:vetor[1..100] de inteiro
  elementos[1..2] de inteiro
  i,j,menor:inteiro
inicio
  maiorDistancia <- 0 
  para i de 1 ate n - 1 faca
    menor <-i 
    para j de i + 1 ate n faca
      se v[j] < v[menor]  entao
        menor <- j
      fimse
    fimpara
    se menor - i > maiorDistancia entao
      maiorDistancia <- menor - i
      elementos[1] <- v[i]
      elementos[2] <- v[menor]
      volta <- i
    fimse
    aux <- v[i]
    v[i] <- v[menor]
    v[menor] <- aux
  fimpara
  escreva("a maior distancia de troca aconteceu entre os elementos " ,elementos[1],elementos[2],", na volta", volta,", e foi de ",maiorDistancia)
fimalgoritmo
