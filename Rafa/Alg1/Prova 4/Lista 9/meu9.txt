algoritmo media_das_distancias_de_todas_as_subidas_no_bubble_sort
const 
  N = 100
var
  v:vetor[1..N] de inteiro
  i,distancias,distancia_atual: inteiro
  trocou:logico
inicio
  trocou <- VERDADEIRO
  distancias <- 0
  enquanto trocou faca
    trocou <- FALSO
    distancia_atual <- 0
    elemento_atual <- 0
    para i de 1 ate n - 1 faca
      se v[i] > v[i+1] entao 
        se v[i] != elemento_atual entao
          distancias <- distancias + distancia_atual
          elemento_atual <- v[i]
          distancia_atual <- 1
        senao
          distancia_atual <- distancia_atual + 1
        fimse
        aux <- v[i]
        v[i] <- v[i+1]
        v[i+1] <- aux
        trocou <- VERDADEIRO
      senao
        distancias <- distancias + distancia_atual
        distancia_atual <- 0
        elemento_atual <- 0
      fimse
    fimpara
    distancias <- distancias + distancia_atual
  fimenquanto
  escreva("a media das distancias de todas as subidas no bubble sort foi de ",distancias / (n-1))
fimalgoritmo
