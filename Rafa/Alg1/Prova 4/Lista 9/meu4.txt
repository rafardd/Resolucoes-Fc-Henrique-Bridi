algoritmo trocas_insertion_sort
var
  v:vetor[1..100] de inteiro
  i,j,aux,trocas:inteiro
inicio
  trocas <- 0
  para i de 2 ate n faca
    aux <- v[i]
    j <- i - 1 
    enquanto j > 0  e v[j] > aux faca
      v[j + 1] <- v[j]
      trocas <- trocas + 1
      j <- j - 1 
    fimenquanto
    v[j + 1] <- aux
  fimpara
  escreva("o insertions sort realizou um total de ",trocas," trocas")
fimalgoritmo
