algoritmo quantas_trocas_selection_sort
var
  N,i,j,menor,aux,trocas: inteiro
  v:vetor[1..100] de inteiro
incio
  trocas <- 0 
  para i de 1 ate n-1 faca
    menor <- i 
    para j de i + 1  ate n faca
      se v[j] < v[menor] entao
        menor <- j
      fimse
    fimpara
    se menor != i entao 
      trocas <- trocas + 1 
    fimse
    aux <- v[i]
    v[i] <- v[menor]
    v[menor] <- aux
  fimpara
  escreva(trocas)
fimalgoritmo
