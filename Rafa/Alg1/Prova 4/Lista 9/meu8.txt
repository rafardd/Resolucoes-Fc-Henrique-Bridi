algoritmo maior_distancia_subida_bubble_sort
const
  N = 100
var
  v:vetor[1..N] de inteiro
  i:inteiro
  trocou:logico
inicio
  trocou <- VERDADEIRO
  maiorDistancia<- 0
  volta <- 0
  enquanto trocou faca
    trocou <- FALSO
    elemento_atual <- 0
    volta <- volta +1
    distancia_atual <- 0
    para i de 1 ate N-1 faca
      se v[i] > v[i+1] entao
        se v[i] != elemento_atual entao
          se distancia_atual > maiorDistancia entao
            maiorDistancia <- distancia_atual
            elemento_recorde<- elemento_atual
            maior_volta <- volta
          fimse
          distancia_atual <- 1
          elemento_atual<- v[i]
        senao
          distancia_atual <- distancia_atual + 1 
        fimse
        aux <- v[i]
        v[i] <- v[i+1]
        v[i+1] <- aux
        trocou  <-  VERDADEIRO
      senao
        se distancia_atual > maiorDistancia entao
          maiorDistancia <- distancia_atual
          elemento_recode <- elemento_atual
          maior_volta <- volta
        fimse
        distancia_atual <- 0
    fimpara
    se distancia_atual > maiorDistancia entao
      maiorDistancia <- distancia_atual
      elemento_recode <- elemento_atual
      maior_volta <- volta
    fimse
  fimenquanto
  escreva("A maior distancia de subida aconteceu com o elemento ",elemento_recorde,", na volta ",maior_volta"," e foi de ",maiorDistancia)
fimalgoritmo
