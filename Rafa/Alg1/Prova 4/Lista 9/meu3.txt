algoritmo media_das_distancias_selection_sort
var
  v:vetor[1..100] de inteiro
  i,j,menor,distancias,aux:inteiro
inicio
  distancias <- 0
  para i de 1 ate n - 1 faca
    menor <- i
    para j de i + 1 ate n faca
      se v[j] < v[menor] entao
        menor <- j
      fimse
    fimpara
    distancias <- distancias + menor - i
    aux <- v[i] 
    v[i] <- v[menor]
    v[menor] <- aux
  fimpara
  escreva("a media das distancias de todas as trocas no selection sort foi de ",distancias/(n - 1))
fimalgoritmo
