algoritmo verifica_vetores_linearmente_dependentes
var
    A,B: vetor[1..39] de real
    i, primeiro_nao_zero: inteiro
    escalar: real
    dependentes: logico
inicio
    para i de 1 ate 39 faca
        leia(A[i])
    fimpara
    para i de 1 ate 39 faca
        leia(B[i])
    fimpara
    escalar <- 0
    primeiro_nao_zero <- 0
    i <- 1
    enquanto i <= 39 e primeiro_nao_zero = 0 faca
        se B[i] != 0 entao
            primeiro_nao_zero <- i
        fimse
        i <- i + 1
    fimenquanto
    se primeiro_nao_zero > 0 entao
        escalar <- A[primeiro_nao_zero] / B[primeiro_nao_zero]
    fimse
    dependentes <- VERDADEIRO
    i <- 1
    enquanto i <= 39 e dependentes faca
        se B[i] = 0 entao
            se A[i] != 0 entao
                dependentes <- FALSO
            fimse
        senao
            se A[i] != escalar * B[i] entao
                dependentes <- FALSO
            fimse
        fimse
        i <- i + 1
    fimenquanto
    se dependentes entao
        escreva("Vetores sao linearmente dependentes")
    senao
        escreva("Vetores nao sao linearmente dependentes")
    fimse
fimalgoritmo
