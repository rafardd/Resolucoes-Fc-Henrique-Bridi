algoritmo qual_a_maior_soma_dos_simetricos
var
  A: vetor [1..128] de real
  i,posicao:inteiro
  maiorSoma:real
inicio
  para i de 1 ate 128 faca
    escreva("insira o valor do vetor A na posicao ",i)
    leia(A[i])
  fimpara
  maiorSoma <- A[1] + A[128]
  posicao <- 1
  para i de 2 ate 64 faca
    se A[i] + A[128 - i + 1] > maiorSoma entao
      maiorSoma <- A[i] + A[128 - i + 1]
      posicao <-i
    fimse
  fimpara
  escreva("a maior soma teve o valor de "maiorSoma)
  escreva("e ocorreu entre as posicoes",posicao,128 - posicao + 1)
fimalgoritmo
