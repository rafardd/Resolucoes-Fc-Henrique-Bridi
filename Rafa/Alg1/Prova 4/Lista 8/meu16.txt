algoritmo ordenacao_par_e_impar
var
  v: vetor [1..153] de inteiro
  i,atual:inteiro
  atual_par,anterior_impar: logico
inicio
  para i de 1 ate 153 faca
    escreva("insira o valor para a posição ",i)
    leia(v[i])
  fimpara
  trocou <- VERDADEIRO
  enquanto trocou faca
    trocou <- FALSO
    para i de 1 ate 152 faca
      atual_impar <- v[i] mod 2 != 0
      proximo_par <- v[i+1] mod 2 = 0
      se atual_impar e proximo_par entao
        atual <- v[i]
        v[i] <- v[i+1]
        v[i+1] <- atual
        trocou <- VERDADEIRO
      fimse
  fimenquanto
fimalgoritmo
