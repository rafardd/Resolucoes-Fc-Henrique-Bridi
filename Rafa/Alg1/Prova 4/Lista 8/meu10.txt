algoritmo media_dos_53_numeros
var
  numeros:vetor[1..53] de inteiro
  i,soma,acima_ou_igual_media:inteiro
  media:real
inicio
  soma <- 0
  para i de 1 ate 53 faca
    escreva("insira um numero inteiro")
    leia(numeros[i])
    soma <- soma + numeros[i]
  fimpara
  media <- soma/53
  acima_ou_igual_media <- 0 
  para i de 1 ate 53 faca
    se numeros[i] >= media entao
      acima_ou_igual_media <- acima_ou_igual_media + 1
    fimse
  fimpara
  escreva(acima_ou_igual_media)
fimalgoritmo
