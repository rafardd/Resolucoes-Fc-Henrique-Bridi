algoritmo x_em_a
const
  MAX:150
  MIN:15
var
  a:vetor[1..150] de inteiro
  n,x:inteiro
inicio
  n <- trunc((MAX - MIN +1)*aleatorio()) mod (MAX - MIN + 1) + MIN
  escreva("insira os valores da sequencia A: ")
  para i de 1 ate n faca
    leia(a[i])
  fimpara
  escreva("insira o valor de x: ")
  leia(x)
  i <- 1 
  meio <- (i+n) div 2 
  enquanto  i <= n e a[meio] != x faca
    se a[meio] > x faca
      n <- meio -1 
    senao faca
      i <- meio + 1
    fimse
    meio <-( i+n) div 2 
  fimenquanto
  se a[meio] = x entao 
      Escreva(meio)
  Senao
       Escreva("0")
