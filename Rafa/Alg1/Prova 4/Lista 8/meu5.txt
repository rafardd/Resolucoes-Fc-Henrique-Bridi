algoritmo frequencia_dos_numeros
var
  frequencia:vetor [1..63] de inteiro
  i,n: inteiro
inicio
  para i de 1 ate 63 faca
    frequencia[i] <- 0
  fim para
  para i de 1 ate 4095 faca
    escreva("insira um numero entre 1 e 63")
    leia(n)
    enquanto 1 > n ou n > 63 faca
      escreva("o numero deve estar entre 1 e 63")
      leia(n)
    fimenquanto
    frequencia[n] <- frequencia[n] + 1 
  fim para
  para i de 1 ate 63 faca
    se frequencia[i] > 0 entao
      escreva("frequencia do numero ",i,":",frequencia[i])
    fimse
  fimpara
fimalgoritmo
