7ª Questão:
algoritmo CalculaFrequênciaDeCaracteres

var
   frequência : vetor[32..126]
   s : string
   n, i : inteiro

início 
   escreva("Insira uma frase: ")
   leia(s)
   se tamanho(s) > 0 então
      para i de 32 até 126 faça
         frequência[i] <- 0
      fimpara
      para i de 1 até tamanho(s) faça
         n <- asc(s[i])
         frequência[n] <- frequência[n] + 1
      fimpara
      para i de 32 até 126 faça
         se frequência[i] > 0 então
            escreva("O caracter ", chr(i), " aparece ", frequência[i], " vez(es)")
         fimse
      fimpara
   senão
      escreva("String vazia")
   fimse
fimalgoritmo
