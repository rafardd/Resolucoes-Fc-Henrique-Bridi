algoritmo n_numeros_na_ordem_inversa
var
  n,i:inteiro
  numeros:vetor[1..111] de real
inicio
  escreva("insira o numero n")
  leia(n)
  enquanto n < 1 ou n > 111 faca
    escreva("deve estar entre 1 e 111")
    leia(n)
  fimenquanto
  para i de 1 ate n faca
    leia(numeros[i])
  fimpara
  para i de 1 ate n faca
    escreva(numeros[n-i+1])
  fimpara
fimalgoritmo
