11ª Questão:
algoritmo EscreveSequênciaNaOrdemInversa

var
   v : vetor[1..111] de real
   n, i : inteiro

início
   escreva("Insira um número entre 1 e 111: ")
   leia(n)
   enquanto n < 1 ou n > 111 faça
      escreva("Inválido. Número deve estar entre 1 e 111")
      leia(n)
   fimenquanto
   para i de 1 até n faça
      escreva("Insira um número real da sequência: ")
      leia(v[i])
   fimpara
   para i de 1 até n faça
      escreva(v[n - i + 1])
   fimpara
fimalgoritmo
