algoritmo posicao_da_diferenca_de_A_B
var
  A,B: vetor[1..320] de inteiro
  i,n: inteiro
inicio
  n <- -1
  enquanto nao(n > 0 e n < 321) faca
    escreva("insira um numero entre 1 e 320")
    leia(n)
  fimenquanto
  para i de 1 ate n faca
    leia(A[i])
  fimpara
  para i de 1 ate n faca
    leia(B[i])
  fimpara
  i <- 1
  enquanto i <= n e A[i] = B[i] faca
    i <- i + 1
  fimenquanto
  se i > n entao
    escreva("sao dois vetores iguais")
  senao
    escreva("a primeira diferenca ocorre na posicao: ",i)
  fimse
fimalgoritmo
