algoritmo CalculaMaiorSomaDeSimétricos

var
   v : vetor[1..128] de real
   i, posição : inteiro
   maior : real

início
   para i de 1 até 128 faça
      escreva("Insira um número real da sequência: ")
      leia(v[i])
   fimpara
   maior <- v[64] + v[65]
   posição <- 64
   para i de 1 até 63 faça
      se v[64 - i + 1] + v[64 + i] >= maior então
         maior <- v[64 - i + 1] + v[64 + i]
         posição <- 64 - i + 1
      fimse
   fimpara
   escreva("A maior soma de simétricos foi ", maior, " e estava nas posições ", posição, " e ", 128 - posição + 1)
