procedimento escreveSequencia(final:inteiro;comeco:inteiro;divisor:inteiro)
inicio
  se comeco <= final entao
    enquanto comeco <= final faca
      se comeco mod divisor = 0 entao
        escreva(comeco)
      fimse
      comeco <- comeco + 1
    fimenquanto
  senao
    enquanto comeco >= final faca
      se comeco mod divisor = 0 entao
        escreva(comeco)
      fimse
      comeco <- comeco - 1
    fimenquanto
  fimse
fimprocedimento
