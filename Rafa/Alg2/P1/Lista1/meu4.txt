procedimento leOsValores(ref s1:string;ref s2:string)
var
  copia : string
inicio
  escreva("insira os dois valores(strings): ")
  leia(s1,s2)
  enquanto s1 = s2 faca
    escreva("os dois valores devem ser diferentes, reeinsira os dois valores: ")
    leia(s1,s2)
  fimenquanto
  se s1 > s2 entao
    copia <- s1
    s1 <- s2
    s2 <- copia
  fimse
fimprocedimento
