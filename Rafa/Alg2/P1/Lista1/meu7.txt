funcao converteStringParaNumero(s:string):inteiro
var
  numero:inteiro
inicio
  numero <- 0
  para i de 1 ate tamanho(s) faca
    se s[i] >= '0' e s[i] <= '9' entao
      numero <- numero * 10
      numero <- numero + asc(s[i]) - asc('0')
    senao
      retorne -99999999
    fimse
  fimpara
  retorne numero
fimfuncao
