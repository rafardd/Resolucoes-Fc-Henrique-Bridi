funcao stringMinuscula(s:string):string
var
  procedimento minimiza(n:inteiro)
  inicio
    se n > 0 entao
      se s[n] >= 'A' e s[n] <= 'Z' entao
        s[n] <- chr(asc(s[n]) + asc('a') - asc('A'))
      fimse
      minimiza(n-1)
    fimse
  fim procedimento
inicio
  minimiza(tamanho(s))
  retorne s
fim funcao
