  funcao stringValida(n:inteiro):logico
  inicio
    se n = 0 entao
      retorne VERDADEIRO
    fimse
    se n = 1 e (decimal[n] = '-' ou decimal[n] = '+') entao
      retorne VERDADEIRO
    fimse
    retorne stringValida(n-1) e (decimal[n] >= '0' e decimal[n] <= '9')
  fim funcao
  
  funcao converteStringParaInt(n:inteiro):inteiro
  inicio
    se n = 0 entao
      retorne 0
    fimse
    se n = 1 e (decimal[n] = '-' ou decimal [n] = '+') entao
      retorne 0
    fimse
    retorne converStringParaInt(n-1) * 10 + asc(decimal[n]) - asc('0')
  fim funcao

  funcao converteDecimalParaBinario(n:inteiro):inteiro
  inicio
    se n = 0 entao
      retorne 0
    senao
      retorne converteDecimalParaBinario(n div 2) * 10 + n mod 2
    fimse
  fim funcao
