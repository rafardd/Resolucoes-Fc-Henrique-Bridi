  funcao stringValida(n:inteiro):logico
  inicio
    se n = 0 entao
      retorne VERDADEIRO
    fimse
    se n = 1 e (octal[n] = '-' ou octal[n] = '+') e tamanho(octal) > 1 entao
      retorne VERDADEIRO
    fimse
    caracterÉValido<- octal[n]  <= '7' e octal[n]>= '0'
    retorne caracterÉValido e stringValida(n-1)
  fim funcao

      retorne 0
    fimse
    retorne conversaoDeBase(n-1,potencia * 8)+asc(octal[n])-asc('0')*potencia
  fim funcao
