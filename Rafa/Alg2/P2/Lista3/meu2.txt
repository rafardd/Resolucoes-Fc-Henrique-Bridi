funcao reverteNumero(n:inteiro):inteiro
var
  funcao fazInverso(n,acumulador:inteiro):inteiro
  inicio
    se n = 0 entao
      retorne acumulador
    senao
      acumulador <- acumulador * 10 + n mod 10
      retorne fazInverso(n div 10,acumulador)
    fimse
  fim funcao
inicio
  se n < 0 entao
    retorne -(fazInverso(abs(n),0))
  senao 
    retorne fazInverso(n,0)
  fimse
fim funcao
