  funcao stringValida(n:inteiro):logico

  funcao stringParaInt(n,potencia:inteiro):inteiro

    se n = 0 entao
      retorne 0
    fimse
    se n = 1 e (hexa[n] = '-' ou hexa[n] = '+') entao
      retorne 0
    fimse
    se hexa[n] >= 'A' entao
      retorne stringParaInt(n-1,potencia * 16) + (asc(hexa[n]) - asc('A') + 10) * potencia
    senao
      retorne stringParaInt(n-1,potencia*16) + (asc(hexa[n]) - asc('0')) * potencia
    fimse
  fim funcao

  funcao intParaString(n:inteiro):string
  inicio
    se n = 0 entao
      retorne ""
    fimse
    retorne intparaString(n div 10) + chr(asc('0') + n mod 10)
  fim funcao
