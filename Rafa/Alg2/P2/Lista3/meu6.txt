tipos
  vet = vetor[1..1000] de string

funcao menorStringLexicograficamente(v:vet;t:inteiro):string
var
  menor:string
inicio
  se t < 1 entao
    pare
  fimse
  se t = 1 entao
    retorne v[t]
  fimse
  menor <- menorStringLexicograficamente(v,t-1)
  se menor < v[t] entao
    retorne menor
  fimse
  retorne v[t]
fim funcao
