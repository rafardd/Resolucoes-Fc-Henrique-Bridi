tipos 
	Pno = ponteiro para Tno
	Tno = registro
		info : real
		chave : inteiro
		prox : Pno
	      fim registro
funcao contemX(l:pno;x:real):logico
inicio
	se l = NULO entao
		retorne FALSO
	fimse
	se l^.info = x entao
		retorne VERDADEIRO	
	fimse
	se l^.info < x entao
		retorne contemX(l^.prox,x)
	fimse
	retorne FALSO
fim funcao
