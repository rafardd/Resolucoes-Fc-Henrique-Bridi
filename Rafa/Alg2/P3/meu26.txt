tipos
	Pno:ponteiro para Tno
	Tno:registro
		info:real
		prox:Pno
	   registro
funcao juntaPorPosicao(ref l:Pno;p:inteiro):logico
var
	temp,aux:Pno
inicio
	se l = NULO entao
		retorne FALSO	
	fimse
	temp <- l
	enquanto temp^.prox != NULO e p > 1 faca
		temp <- temp^.prox
		p <- p - 1
	fim enquanto
	se temp^.prox = NULO entao
		retorne FALSO
	fimse
	temp^.info <- temp^.info + temp^.prox^.info
	aux <- temp^.prox
	temp^.prox <- temp^.prox^.prox
	desaloca(aux)
	retorne VERDADEIRO
fim funcao
