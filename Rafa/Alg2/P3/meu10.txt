tipos
	Pno:ponteiro para Tno
	Tno: registro 
		info: real
		chave: inteiro
		prox : Pno
	    fim registro
funcao aparecemNaLista(l:Pno;x,y:real):logico
var
	apareceX,apareceY:logico
inicio
	apareceX <- FALSO
	apareceY <- FALSO
	enquanto l != NULO faca
		se l^.info = x entao
			apareceX <- VERDADEIRO
		senao
			se l^.info = y e apareceX entao
				apareceY <- VERDADEIRO
			fimse
		fimse
		l <- l^.prox
	fim enquanto
	retorne apareceX e apareceY
fim funcao
