tipos
	Pno:ponteiro para Tno
	Tno:	registro
			info : real
			chave : inteiro
			prox : Pno
		fim registro
funcao contemX(l:Pno;x:real):logico
inicio
	enquanto l != NULO e l^.info < x faca
		l <- l^.prox
	fim enquanto
	se l!= NULO e l^.info = x entao
		retorne VERDADEIRO
	fimse
	retorne FALSO
fim funcao
