tipos
	Pno:ponteiro para Tno
	Tno:registro
		info:inteiro
		prox:Pno
funcao insereOrdenada(ref l:Pno;x:inteiro):logico
var
	novo:Pno
inicio	
	se l = NULO ou l^.info > x  entao
		novo <- aloca(Tno)
		se novo = NULO entao
			retorne FALSO
		fimse
		novo^.info <- x
		novo^.prox <- l
		l <- novo
		retorne VERDADEIRO
	fimse
	retorne insereOrdenada(l^.prox,x)
fim funcao
