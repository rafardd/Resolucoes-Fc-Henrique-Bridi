tipos
	Pno:ponteiro para Tno
	Tno: registro 
		info: real
		chave: inteiro
		prox : Pno
	    fim registro
funcao aparecemJuntos(l:Pno;x,y:real):logico
var
	
inicio
	enquanto l!= NULO faca
		se l^.info = x entao
			se l^.prox != NULO e l^.prox^.info = y entao
				retorne VERDADEIRO
			fim se

		fimse
		l <- l^.prox
	fim enquanto
	retorne FALSO
fim funcao
