tipos
	Pno:ponteiro para Tno
	Tno:registro
		info:string
		prox:Pno
		ant:Pno
	    fim registro
funcao removePrimeiraOcorrencia(ref l:Pno;x:string):logico
var
	aux:Pno
inicio
	se l = NULO entao
		retorne FALSO	
	fimse
	se l^.info = x entao
		aux <- l
		se l^.prox != NULO entao
			l^.prox^.ant <- l^.ant
		fimse
		l <- l^.prox
		desaloca(aux)
		retorne VERDADEIRO
	fimse
	retorne removePrimeiraOcorrencia(l^.prox,x)
fim funcao
