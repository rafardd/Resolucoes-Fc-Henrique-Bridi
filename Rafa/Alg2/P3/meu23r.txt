tipos
	Pno:ponteiro para Tno
	Tno:registro
		info:string
		prox:Pno
	    fim registro
funcao removeMaioresOuIguais(ref l:Pno;x:string):logico
var
	temp:Pno
inicio
	se l = NULO entao 
		retorne FALSO
	fimse
	se l^.info >= x entao
		temp <- l
		l <- l^.prox
		desaloca(temp)
		retorne removeMaioresOuIguais(l,x) ou VERDADEIRO
	fimse
	retorne FALSO ou removeMaioresOuIguais(l^.prox,x)
fim funcao
