tipos
	Pno:pointeiro para Tno
	Tno:registro
		info:string
		prox:Pno
	   fim registro
funcao removeMaoiresOuIguais(ref l:Pno;x:string):logico
var
	p,temp:Pno
	removeu:logico
inicio
	removeu <- FALSO
	enquanto l!= NULO e l^.info >= x faca
		temp <- l
		l <- l^.prox
		desaloca(temp)
		removeu <- VERDADEIRO
	fim enquanto
	se l = NULO entao
		retorne removeu
	fimse
	p <- l
	enquanto p^.prox != NULO faca
		se p^.prox^.info >= x entao
			temp <- p^.prox
			p^.prox <- p^.prox^.prox
			desaloca(temp)
			removeu <- VERDADEIRO
		senao
			p <- p^.prox
		fimse
	fim enquanto
	retorne removeu
fim funcao
