tipos
	Pno:ponteiro para Tno
	Tno: registro 
		info: real
		chave: inteiro
		prox : Pno
	    fim registro
funcao aparecemJuntos(l:Pno;x,y:real):logico
inicio	
	se l = NULO entao
		retorne FALSO
	fim se
	se l^.info = x entao
		se l^.prox != NULO e l^.prox^.info = y entao
			retorne VERDADEIRO
		fimse
	fim se
	retorne aparecemJuntos(l^.prox,x,y)
fim funcao
