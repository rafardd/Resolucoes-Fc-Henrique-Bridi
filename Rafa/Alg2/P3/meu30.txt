tipos
	Pno:ponteiro para Tno
	Tno:registro
		coef:real
		exp:real
		prox:Pno
	    fim registro
funcao somaPolinomios(p1,p2:Pno):Pno
var
	funcao criaNo(e,c):logico
	var
		aux,novo:Pno	
	inicio	
		novo <- aloca(Tno)
		se novo = NULO entao
			enquanto p != NULO faca
				aux <- p 
				p <- p^.prox
				desaloca(aux)
			fim enquanto
			ult <- NULO
			retorne FALSO
		fimse
		novo^.coef <- c
		novo^.prox <- NULO
		novo^.exp <- e
		se p = NULO entao
			p <- novo
		senao
			ult^.prox <- novo
		fimse
		ult <- novo
		retorne VERDADEIRO
	fim funcao
	p,ult:Pno
inicio
	p <- NULO
	ult <- NULO
	enquanto p1 != NULO e p2 != NULO faca
		se p1^.exp = p2^.exp entao
			coef <- p1^.coef + p2^.coef
			se coef != 0 entao
				se nao criaNo(p1^.exp,coef) entao
					retorne NULO
				fimse
			fimse
			p1 <- p1^.prox
			p2 <- p2^.prox
		senao
			se p1^.exp > p2^.exp entao
				se nao criaNo(p1^.exp,p1^.coef) entao
					retorne NULO
				fimse
				p1 <- p1^.prox
			senao
				se nao criaNo(p2^.exp,p2^.coef) entao
					retorne NULO
				fimse
				p2 <- p2^.prox
			fimse
		fimse
	fim enquanto
	enquanto p1!= NULO faca
		se nao criaNo(p1^.exp,p1^.coef) entao
			retorne NULO
		fimse
		p1 <- p1^.prox
	fim enquanto
	enquanto p2 != NULO faca
		se nao criaNo(p2^.exp,p2^.coef) entao
			retorne NULO
		fimse
		p2 <- p2^.prox
	fim enquanto
	retorne p
fim funcao
