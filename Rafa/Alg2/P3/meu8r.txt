tipos
	Pno:ponteiro para Tno
	Tno: registro 
		info: string
		chave: inteiro
		prox : Pno
	    fim registro
funcao maiorString(l:Pno):string
var
	maior : string
inicio
	se l = NULO entao
		retorne ""
	fimse
	maior <- maiorString(l^.prox)	
	se l^.info > maior entao
		retorne l^.info
	fimse
	retorne maior
fim funcao
