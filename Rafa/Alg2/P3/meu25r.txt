tipos
	Pno:ponteiro para TNo
	Tno:registro
		info:real
		prox:Pno
procedimento ultimoParaPrimeiro(ref l:Pno)
var
	temp: Pno
inicio
	se l!=NULO e l^.prox!= NULO entao
		se l^.prox^.prox = NULO entao
			temp <- l^.prox
			l^.prox <- NULO
			temp^.prox <- l
			l <- temp
		senao
			ultimoParaPrimeiro(l^.prox)
			temp <-l^.prox
			l^.prox <- l^.prox^.prox
			temp^.prox <- l
			l <- temp
		fimse
	fimse
fim procedimento
