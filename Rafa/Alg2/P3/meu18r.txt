tipos
	Pno:ponteiro para Tno
	Tno:registro
		info:real
		prox:Pno
funcao inserePorPosicao(ref l:Pno;x:real;p:inteiro):logico
var
	novo:Pno
inicio
	se p < 1 entao
		retorne FALSO
	fimse
	se l = NULO e p > 1 entao
		retorne FALSO
	fimse
	se p = 1 entao
		novo <- aloca(Tno)
		se novo = NULO entao
			retorne FALSO
		fimse
		novo^.info <- x
		novo^.prox <- l
		l <- novo
		retorne VERDADEIRO
	fimse
	retorne inserePorPosicao(l^.prox,x,p-1)
fim funcao
