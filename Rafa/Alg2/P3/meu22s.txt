tipos
	Pno:ponteiro para Tno
	Tno:registro
		info:string
		prox:Pno
funcao removePrimeiraOcorrencia(ref l:Pno;x:string):logico
var
	temp,aux:Pno
inicio
	se l = NULO entao
		retorne FALSO
	fimse
	temp <- l
	se l^info = x entao
		l <- l^.prox
		desaloca(temp)
		retorne VERDADEIRO
	fimse
	enquanto temp^.prox != NULO e temp^.prox^.info != x faca
		temp <- temp^.prox
	fim enquanto
	se temp^.prox != NULO entao
		aux <- temp^.prox
		temp^.prox <- temp^.prox^.prox
		desaloca(aux)
		retorne VERDADEIRO
	fimse
	retorne FALSO
fim funcao
