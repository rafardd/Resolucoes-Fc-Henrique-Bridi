tipos
	Pno:ponteiro para Tno
	Tno:registro
		info:inteiro
		prox:Pno
	   fim registro
procedimento paresImpares(l:Pno;ref lpar,limpar:Pno)
var
	ppar,pimpar,novo:Pno
	procedimento limpaLista(l:Pno)
	var
		temp <- Pno
	inicio
		enquanto l != NULO faca
			temp <- l
			l <- l^.prox
			desaloca(temp)
		fim enquanto
	fim procedimento
inicio
	ppar <- lpar
	pimpar <- limpar
	enquanto l != NULO faca
		novo <- aloca(Tno)
		se novo = NULO entao
			limpaLista(lpar)
			limpaLista(limpar)
			pare
		fimse
		novo^.info <- l^.info
		novo^.prox <- NULO
		se novo^.info mod 2 = 0 entao
			se lpar = NULO entao
				lpar <- novo
				ppar <- novo
			senao
				ppar^.prox <- novo
				ppar <- ppar^.prox
			fimse
		senao
			se limpar = NULO entao
				limpar <- novo
				pimpar <- novo
			senao
				pimpar^.prox <- novo
				pimpar <- pimpar^.prox
			fimse
		fimse
 		l <- l^.prox
	fim enquanto
fim procedimento
