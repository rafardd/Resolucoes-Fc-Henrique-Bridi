funcao removePorPosicao(ref l:Pno;p:inteiro):logico
var
	temp:Pno
inicio
	se p < 1 ou l = NULO entao
		retorne FALSO
	fimse
	se p = 1 entao
		temp <- l
		l <- l^.prox
		desaloca(temp)
		retorne VERDADEIRO
	fim se
	retorneremoverPorPosicao(l^.prox,p-1)
fim funcao
