tipos
	Pno:ponteiro para Tno
	Tno:registro
		info:string
		prox:Pno
	    fim registro
funcao removePrimeiraOcorrencia(ref l:Pno;x:string):logico
var
	aux:Pno
inicio
	se l = NULO entao
		retorne FALSO
	fimse
	se l^.info = x entao
		aux <- l
		l <- l^.prox
		desaloca(aux)
		retorne VERDADEIRO
	fim se
	retorne removePrimeiraOcorrencia(l^.prox,x)
fim funcao
