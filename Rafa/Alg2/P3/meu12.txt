tipos 
	Pno: ponteiro para Tno
	Tno : registro
		info: string
		chave:inteiro
		prox: Pno
	      fim registro
funcao	saoIguais(l1,l2:Pno):logico
inicio
	enquanto l1 != NULO e l2 != NULO e l1^.info = l2^.info faca
		l1 <- l1^.prox
		l2 <- l2^.prox
	fim enquanto		
	retorne l1 = NULO e l2 = NULO
fim funcao
