tipos
	Pno:ponteiro para Tno
	Tno:registro
		info:inteiro
		prox:Pno
	    fim registro
procedimento paresEImpares(l:Pno;ref lPar,lImpar:Pno)
var
	novo:Pno
inicio
	se l != NULO entao
		novo <- aloca(Tno)
		se novo != NULO entao
			novo^.info <- l^info
			se novo^.info mod 2 = 0 entao
				lPar <- novo
				paresEImpares(l^.prox,lpar^.prox,limpar)
			senao
				lImpar <- novo
				paresEImpares(l^.prox,lpar,limpar^.prox)
			fimse
		senao
			pare
		fimse
	senao
		lImpar <- NULO
		lPar <- NULO
	fimse
fim procedimento
