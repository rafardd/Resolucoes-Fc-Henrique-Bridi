tipo 
    PNo : ponteiro para TNo
    TNo : registro
            info : inteiro
            prox, prev : PNo
        fimregistro 

funcao removePrimeiraOcorrenciaDeXDupla(ref head : PNo; x : inteiro) : logico
var
    aux : PNo
inicio
    se head = NULO entao 
        retorne FALSO
    fimse
     
    se head^.info = x entao
        aux <- head
        head <- aux^.prox
        se head != NULO entao
            head^.prev <- NULO
        fimse
        desaloca(aux)
        retorne VERDADEIRO
    fimse

    aux <- head
    enquanto aux != NULO e aux^.info != x faca
        aux <- aux^.prox
    fimenquanto

    se aux = NULO entao
        retorne FALSO
    fimse

    aux^.prev^.prox <- aux^.prox
    # Aqui, quem mudou foi o campo PROX da variavel dinamica que AUX aponta. Mas aux ainda se refere a mesma variavel dinamica.
    se aux^.prox != NULO entao
        aux^prox^.prev <- aux^.prev
    fimse
    desaloca(aux)
    retorne verdadeiro

fimfuncao
