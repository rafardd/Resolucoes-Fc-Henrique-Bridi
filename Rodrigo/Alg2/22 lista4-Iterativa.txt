tipo 
    PNo : ponteiro para TNo
    TNo : registro
            info : inteiro
            prox : PNo
        fimregistro 

funcao removePrimeiraOcorrenciaDeX(ref head : PNo; x : inteiro) : logico
var
    aux, ant : PNo
inicio
    se head = NULO entao
        retorne falso
    fimse

    se head^.info = x entao
        aux <- head
        head <- aux^.prox
        desaloca(aux)
        retorne verdadeiro
    fimse


    ant <- head
    aux <- head^.prox
    enquanto aux != NULO e aux^.info != x faca
        ant <- aux
        aux <- aux^.prox
    fimenquanto
    
    se aux = NULO entao
        retorne falso
    fimse

    ant^.prox <- aux^.prox
    desaloca(aux)
    retorne verdadeiro

fimfuncao
