tipo 
    PNo = ponteiro para TNo
    TNo = registro
            info : string
            prox : PNo
        fimregistro

funcao elementoCentralListaEncadeada(head : PNo) : string
var
    p : PNo
    tam, i, limite : inteiro
inicio
    se head = NULO entao
        retorne ""
    senao
        p <- head
        tam <- 0
        enquanto p != NULO faca
            tam <- tam + 1
            p <- p^.prox
        fimenquanto
        se tam = 1 entao
            retorne head^.info
        senao 
            i <- 1
            limite <- (tam - 1) div 2
            enquanto i <= limite faca   
                head <- head^.prox
                i <- i + 1
            fimenquanto
            retorne head^.info
        fimse
    fimse
fimfuncao
