tipo
    PNo : ponteiro para TNo 
    TNo : registro
            info : inteiro
            prox : PNo 
        fimregistro

funcao verificaXeYNaListaEmOrdem(head : PNo; x, y : inteiro) : logico
var
    achouX, achouY : logico
inicio
    achouX <- falso
    achouY <- falso
    se head = NULO entao
        retorne falso
    fimse 
    enquanto head != NULO e nao achouX faca
        achouX <- achouX ou x = head^.info
        head <- head^.prox
    fimenquanto
    se head = NULO entao
        retorne falso
    senao
        enquanto head != NULO e nao achouY faca
            achouY <- achouY ou y = head^.info
            head <- head^.prox
        fimenquanto
        se achouX e achouY entao
            retorne verdadeiro
        senao
            retorne falso
        fimse
    fimse
fimfuncao
