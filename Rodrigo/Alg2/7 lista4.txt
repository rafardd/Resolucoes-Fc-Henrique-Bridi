            info : inteiro
            prox : PNo

funcao mediaListaEncadeada(head : PNo) : real
var
    media : real
    soma, qntdElementos : inteiro
inicio 
    se head = NULO entao
        retorne 0
    senao
        soma <- 0
        qntdElementos <- 0
        enquanto head != NULO faca
            soma <- soma + head^.info
            qntdElementos <- qntdElementos + 1
            head <- head^.prox
        fimenquanto
        media <- soma / qntdElementos
        retorne media
    fimse
fimfuncao
