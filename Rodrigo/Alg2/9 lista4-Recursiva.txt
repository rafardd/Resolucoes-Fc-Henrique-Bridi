tipo 
    PNo : ponteiro para TNo
    TNo : registro
            info : real
            prox : PNo
        fimregistro

funcao verificaElementosJuntos(head : PNo; x, y : real) : logico
var
    
inicio
    se head = NULO entao    
        retorne falso
    fimse
    se head^.prox != NULO entao
        se head^.info = x e head^.prox^.info = y entao 
            retorne verdadeiro
        senao
            retorne verificaElementosJuntos(head^.prox, x, y)
        fimse
    fimse
    retorne falso
fimfuncao
