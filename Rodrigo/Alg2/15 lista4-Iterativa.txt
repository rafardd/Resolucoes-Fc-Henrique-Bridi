tipo 
    PNo : ponteiro para TNo
    TNo : registro
            info : string
            prox : PNo
        fimregistro

funcao insereNóFinalLista(ref head : PNo; x : string) : logico
var 
    novo, aux : PNo
inicio
    novo <- aloca(TNo)
    se novo = NULO entao 
        retorne falso 
    fimse
    novo^.prox <- NULO
    novo^.info <- x
    se head = NULO entao
        head <- novo
        retorne verdadeiro
    fimse
    aux <- head
    enquanto aux^.prox != NULO faca
        aux <- aux^.prox
    fimenquanto
    aux^.prox <- novo
    retorne verdadeiro
fimfuncao
