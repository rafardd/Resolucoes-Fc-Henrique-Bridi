    procedimento trocaPrimeiroNoComUltimoInterna(ref head, ref headReal : PNo)

        aux : PNo

            pare

            aux <- head^.prox

            aux^.prox <- headReal
            headReal <- aux
        senao
            trocaPrimeiroNoComUltimoInterna(head^.prox, headReal)
        fimse
