tipo 
    PNo : ponteiro para TNo
    TNo : registro
            info : inteiro
            prox : PNo
        fimregistro

funcao insereXNaListaOrdenada(ref head : PNo; x : inteiro) : logico
var
    novo : PNo
inicio
    se head = NULO ou head^.info >= x entao
        se se head != nulo e head^.info = x entao
            retorne falso
        fimse
        novo <- aloca(TNo)
        se novo = NULO entao
            retorne falso
        fimse
        novo^.info <- x 
        novo^.prox <- head
        head <- novo
        retorne verdadeiro
    fimse
    retorne insereXNaListaOrdenada(head^.prox, x)
fimfuncao
