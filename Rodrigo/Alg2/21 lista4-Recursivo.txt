tipo 
    PNo : ponteiro para TNo
    TNo : registro
            info : strings
            prox : PNo
        fimregistro 

funcao removeNoPosicaoP(ref head : PNo, p : inteiro) : logico
var
    aux : PNo
inicio
    se head = NULO ou p <= 0 entao
        retorne falso
    fimse
    se p = 1 entao
        aux <- head
        head <- aux^.prox
        desaloca(aux)
        retorne verdadeiro
    fimse
    retorne removeNoPosicaoP(head^.prox, p - 1)
fimfuncao
