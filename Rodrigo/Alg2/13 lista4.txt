tipo 
    PNo : ponteiro para TNo
    TNo : registro
            info : inteiro
            prox : PNo
        fimregistro

funcao verificaTodosElementosListaEmOutra(head1, head2 : PNo) : logico
var 
    encontrado : logico
    aux : PNo
inicio
    encontrado <- verdadeiro
    enquanto head1 != NULO e encontrado faca
        encontrado <- falso
        aux <- head2
        enquanto aux != NULO e nao encontrado faca
            se aux^.info = head1^.info entao 
                encontrado <- verdadeiro
            fimse 
            aux <- aux^.prox
        fimenquanto
        head1 <- head1^.prox
    fimenquanto
    retorne encontrado
fimfuncao
