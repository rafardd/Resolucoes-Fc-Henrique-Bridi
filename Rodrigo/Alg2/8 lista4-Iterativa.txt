tipo 
    PNo : ponteiro para TNo
    TNo : registro
            info : string
            prox : PNo
        fimregistro

funcao maiorStringEmListaEncadeada(head : PNo) : string
var
    maior : string
inicio
    se head = NULO entao
        retorne ""
    fimse
    maior <- head^.info
    head <- head^.prox
    enquanto head != NULO faca
        se head^.info > maior entao
            maior <- head^.info
        fimse
        head <- head^.prox
    fimenquanto
    retorne maior
fimfuncao
