tipo 
    PNo : ponteiro para TNo
    TNo : registro
            info : inteiro
            prox : PNo
        fimregistro 

procedimento trocaPrimeiroNoComUltimo(ref head : PNo)
var
    aux, ant : PNo
inicio
    se head = NULO ou head^.prox = NULO entao
        pare
    fimse

    ant <- head
    aux <- head^.prox
    enquanto aux^.prox != NULO faca
        ant <- aux
        aux <- aux^.prox
    fimenquanto

    ant^.prox <- NULO
    aux^.prox <- head
    head <- aux


fimprocedimento

# Corto o ultimo, aponto ele para o antigo primeiro e atualizo o head
