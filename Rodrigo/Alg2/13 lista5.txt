Faça o algoritmo de uma função que receba o nome de uma sequência de inteiros, sabendo que os valores
nela contidos variam entre 0 e 3000, e retorne a moda desta sequência. Se a sequência for multimodal,
retornar o maior valor entre os que estão na moda.

abrir o arquivo
gerar o vetor de frequencia
fechar o arquivo
percorrer o vetor de frequencia, guardando a maior frequencia
percorro o vetor de trás pra frente e, achando essa frequencia, retorno.

funcao modaValoresAté3000(nome : string) : inteiro
var 
  v : vetor[0..3000] de inteiro
  maiorFreq, i, x, moda : inteiro
  f : sequencia de inteiro
inicio
  se abre(f, nome) entao
    maiorFreq <- 0
    para i de 0 até 3000 faca
      v[i] <- 0
    fimpara
    enquanto nao fds(f) faca
      leia(f, x)
      v[x] <- v[x] + 1
    fimenquanto
    fecha(f)
    
    para i de 0 até 3000 faca
      se v[i] > maiorFreq entao
        maiorFreq <- v[i]
      fimse
    fimpara

    i <- 3000
    enquanto i >= 0 faca
      se v[i] = maiorFreq entao
        moda <- i
        retorne moda
      fimse
      i <- i - 1
    fimenquanto
  senao
    retorne -1
  fimse
