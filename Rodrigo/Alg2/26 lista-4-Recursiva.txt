tipo 
    PNo : ponteiro para TNo
    TNo : registro
            info : inteiro
            prox : PNo
        fimregistro 

funcao juntarNoPComProximo(ref head : PNo; p : inteiro) : logico
var
    aux : PNo
inicio
    se head = NULO ou head^.prox = NULO entao
        retorne falso
    fimse

    se p = 1 entao
        aux <- head^.prox
        head^.info <- head^.info + aux^.info
        head^.prox <- aux^.prox
        desaloca(aux)
        retorne verdadeiro
    fimse

    retorne juntarNoPComProximo(head^.prox, p - 1)

fimfuncao
