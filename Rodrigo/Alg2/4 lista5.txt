procedimento imprimeTermosAPartirDaMetadeSequencia(nome : string)
var
  i, tamanho : inteiro
  f : sequencia de real
  x, metade : real
inicio
  tamanho <- 0
  se abre(f, nome) entao
    enquanto nao fds(f) faca
      leia(f, x)
      // Tomar cuidado com loops, no contexto de arquivos, que não precisam altera-los/le-los, pois, assim, o apontador não muda de posição e vira um loop infinito!
      tamanho <- tamanho + 1
    fimenquanto
    fecha(f)
  fimse

  se abre(f, nome) entao
    metade <- tamanho div 2
    para i de 1 até tamanho faca
      leia(f, x) 
      // Tomar cuidado com condicionais, dentro de loops, no contexto de arquivos, para nao correr o risco do arquivo não andar por falta de leia() ou escreva()
      se i >= metade + 1 entao
        escreva(x)
      fimse
    fimpara
    fecha(f)
  fimse
fimprocedimento
