tipo 
    PNo : ponteiro para TNo
    TNo : registro
            info : strings
            prox : PNo
        fimregistro 

funcao retiraUltimoNó(ref head : PNo) : logico
var
    aux : PNo
inicio
    se head = NULO entao
        retorne falso
    fimse
    se head^.prox = NULO entao
        aux <- head
        head <- NULO
        desaloca(aux)
        retorne verdadeiro 
    fimse 
    retorne retiraUltimoNó(head^.prox)
fimfuncao
