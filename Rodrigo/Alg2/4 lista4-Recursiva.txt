tipo 
    PNo = ponteiro para TNo
    TNo = registro
            info : real
            prox : PNo
        fimregistro

funcao qntdVezesListaEncadeada(head : PNo; x : real) : inteiro
inicio
    se head = NULO entao
        retorne 0
    fimse
    se head^.info = x entao
        retorne 1 + qntdVezesListaEncadeada(head^.prox, x)
    senao
        retorne qntdVezesListaEncadeada(head^.prox, x)
    fimse
fimfuncao
