tipo
    PNo : ponteiro para TNo 
    TNo : registro
            info : inteiro
            prox : PNo 
        fimregistro

funcao interna(head1, head2 : PNo;) : logico
var
inicio
    se head1 = NULO ou head2 = NULO entao   
        se head1 = head2 entao
            retorne verdadeiro
        senao
            retorne falso
       fimse
    fimse
    se head1^.info = head2^.info entao
       retorne interna(head1^.prox, head2^.prox)
    senao
        retorne falso
    fimse
fimfuncao
